## Task 2: ConnectionPool 통합 리팩터링

**Goal:** Plans 3(Redis), 4(PG)에서 구현한 풀의 공통 패턴을 분석하고, 중복 제거가 가능한 부분을 ConnectionPool CRTP 기반 클래스로 끌어올린다.

**Files:**
- Modify: `apex_shared/lib/adapters/common/include/apex/shared/adapters/connection_pool.hpp`
- 필요 시 Redis/PG 풀 수정

### Steps

- [ ] **Step 1: 공통 패턴 분석**

Redis/PG 풀 구현을 비교하여 다음 항목의 공통화 가능성을 확인:

| 패턴 | RedisPool | PgPool | 공통화 |
|------|-----------|--------|--------|
| min/max pool sizing | PoolConfig 사용 | PoolConfig 사용 | 이미 공통 |
| health_check 타이머 스케줄링 | Asio steady_timer | Asio steady_timer | 추출 가능 |
| shrink_idle 타이머 스케줄링 | Asio steady_timer | Asio steady_timer | 추출 가능 |
| 커넥션 warm-up (min_size까지 사전 생성) | 개별 구현 | 개별 구현 | 추출 가능 |
| 통계 (acquire/release/create/destroy 카운터) | 개별 구현 | 개별 구현 | 추출 가능 |

- [ ] **Step 2: ConnectionPool에 타이머 기반 유지보수 추출 (필요 시)**

두 풀이 동일한 타이머 스케줄링 패턴을 사용한다면 ConnectionPool에 `start_maintenance()` 추가:

```cpp
/// ConnectionPool에 추가 (타이머 스케줄링이 공통 패턴인 경우만)
void start_maintenance(boost::asio::io_context& io_ctx) {
    health_timer_ = std::make_unique<boost::asio::steady_timer>(io_ctx);
    shrink_timer_ = std::make_unique<boost::asio::steady_timer>(io_ctx);
    schedule_health_check();
    schedule_shrink();
}
```

> **주의**: 이 리팩터링은 실제 구현 후 공통 패턴이 확인된 경우에만 수행한다.
> 두 풀의 구현이 충분히 다르다면 (예: Redis는 동기 validate, PG는 비동기 validate), 무리한 공통화보다 현재 구조를 유지하는 것이 YAGNI 원칙에 부합한다.

```cpp
// apex_shared/tests/integration/test_kafka_roundtrip.cpp
#include <apex/shared/adapters/kafka/kafka_adapter.hpp>
#include <apex/shared/adapters/kafka/kafka_producer.hpp>
#include <apex/shared/adapters/kafka/kafka_consumer.hpp>
#include <apex/shared/adapters/kafka/kafka_config.hpp>

#include <gtest/gtest.h>

#include <boost/asio/io_context.hpp>

#include <atomic>
#include <chrono>
#include <string>
#include <thread>

using namespace apex::shared::adapters::kafka;

class KafkaRoundtripTest : public ::testing::Test {
protected:
    void SetUp() override {
        config_.brokers = "localhost:9092";
        config_.consumer_group = "apex-integration-test";
        config_.consume_topics = {"apex-test-roundtrip"};
    }

    KafkaConfig config_;
};

TEST_F(KafkaRoundtripTest, ProduceAndConsume) {
    // Producer 설정 + 초기화
    KafkaProducer producer(config_);
    auto init_result = producer.init();
    ASSERT_TRUE(init_result.has_value())
        << "Producer init failed — is Kafka running on localhost:9092?";

    // 테스트 메시지 전송
    const std::string test_key = "integration-test-key";
    const std::string test_value = "hello-kafka-" +
        std::to_string(std::chrono::steady_clock::now().time_since_epoch().count());

    auto produce_result = producer.produce("apex-test-roundtrip", test_key, test_value);
    ASSERT_TRUE(produce_result.has_value());

    // flush로 delivery 보장
    ASSERT_TRUE(producer.flush(std::chrono::milliseconds{5000}));

    // Consumer로 수신 확인
    boost::asio::io_context io_ctx;
    KafkaConsumer consumer(config_, 0, io_ctx);

    std::atomic<bool> received{false};
    std::string received_value;

    consumer.set_message_callback(
        [&](std::string_view /*topic*/, int32_t /*partition*/,
            std::span<const uint8_t> /*key*/, std::span<const uint8_t> payload,
            int64_t /*offset*/) {
            std::string val(reinterpret_cast<const char*>(payload.data()), payload.size());
            if (val == test_value) {
                received_value = val;
                received.store(true, std::memory_order_release);
            }
        });

    auto consumer_init = consumer.init();
    ASSERT_TRUE(consumer_init.has_value())
        << "Consumer init failed";

    consumer.start_consuming();

    // 최대 10초 대기
    auto deadline = std::chrono::steady_clock::now() + std::chrono::seconds{10};
    while (!received.load(std::memory_order_acquire) &&
           std::chrono::steady_clock::now() < deadline) {
        io_ctx.run_for(std::chrono::milliseconds{100});
    }

    consumer.stop_consuming();

    EXPECT_TRUE(received.load()) << "Message not received within 10 seconds";
    EXPECT_EQ(received_value, test_value);
}
```

```cpp
#include <apex/shared/adapters/redis/redis_adapter.hpp>
#include <apex/shared/adapters/redis/redis_pool.hpp>
#include <apex/shared/adapters/redis/redis_config.hpp>

#include <boost/asio/co_spawn.hpp>
#include <boost/asio/detached.hpp>
#include <boost/asio/io_context.hpp>

using namespace apex::shared::adapters::redis;

class RedisRoundtripTest : public ::testing::Test {
protected:
    void SetUp() override {
        config_.host = "localhost";
        config_.port = 6379;
        config_.pool_size_per_core = 2;
    }

    RedisConfig config_;
    boost::asio::io_context io_ctx_;
};

TEST_F(RedisRoundtripTest, SetAndGet) {
    bool test_passed = false;

    boost::asio::co_spawn(io_ctx_, [&]() -> boost::asio::awaitable<void> {
        RedisPool pool(io_ctx_, config_,
            apex::shared::adapters::PoolConfig{.min_size = 1, .max_size = 4});

        // SET
        auto set_result = co_await pool.set("test:integration:key", "hello-redis");
        EXPECT_TRUE(set_result.has_value());

        // GET
        auto get_result = co_await pool.get("test:integration:key");
        EXPECT_TRUE(get_result.has_value());
        EXPECT_TRUE(get_result->has_value());
        EXPECT_EQ(get_result->value(), "hello-redis");

        // DEL
        auto del_result = co_await pool.del("test:integration:key");
        EXPECT_TRUE(del_result.has_value());
        EXPECT_EQ(del_result.value(), 1);

        // GET after DEL
        auto get2_result = co_await pool.get("test:integration:key");
        EXPECT_TRUE(get2_result.has_value());
        EXPECT_FALSE(get2_result->has_value());  // nullopt

        test_passed = true;
        co_return;
    }, boost::asio::detached);

    io_ctx_.run();
    EXPECT_TRUE(test_passed) << "Redis roundtrip coroutine did not complete";
}

TEST_F(RedisRoundtripTest, SetWithExpire) {
    bool test_passed = false;

    boost::asio::co_spawn(io_ctx_, [&]() -> boost::asio::awaitable<void> {

        co_return;
```

```cpp
            "  id SERIAL PRIMARY KEY,"
            "  name TEXT NOT NULL,"
            "  created_at TIMESTAMPTZ DEFAULT NOW()"
            ")");
        EXPECT_TRUE(create_result.has_value());

        // INSERT
        auto insert_result = co_await conn.execute_async(
            "INSERT INTO integration_test (name) VALUES ('apex-test')");
        EXPECT_TRUE(insert_result.has_value());
        EXPECT_GE(insert_result.value(), 1);

        // SELECT
        auto select_result = co_await conn.query_async(
            "SELECT name FROM integration_test WHERE name = 'apex-test' LIMIT 1");
        EXPECT_TRUE(select_result.has_value());
        EXPECT_TRUE(select_result->ok());
        EXPECT_GE(select_result->row_count(), 1);
        EXPECT_EQ(select_result->value(0, 0), "apex-test");

        // CLEANUP
        co_await conn.execute_async("DROP TABLE IF EXISTS integration_test");

        conn.close();
        test_passed = true;
        co_return;
    }, boost::asio::detached);

    io_ctx_.run();
    EXPECT_TRUE(test_passed);
}

TEST_F(PgRoundtripTest, PoolAcquireRelease) {
    bool test_passed = false;

    boost::asio::co_spawn(io_ctx_, [&]() -> boost::asio::awaitable<void> {
        PgPool pool(io_ctx_, config_,
            apex::shared::adapters::PoolConfig{.min_size = 1, .max_size = 4});

        // acquire → 쿼리 → release
        auto conn_result = pool.acquire();
        EXPECT_TRUE(conn_result.has_value());

        if (conn_result.has_value()) {
            auto& conn = conn_result.value();
            auto query_result = co_await conn->query_async("SELECT version()");
            EXPECT_TRUE(query_result.has_value());
            pool.release(std::move(conn_result.value()));
        }

        EXPECT_EQ(pool.idle_count(), 1u);
        pool.close_all();
        test_passed = true;
        co_return;
    }, boost::asio::detached);

    io_ctx_.run();
    EXPECT_TRUE(test_passed);
}
```

```cpp
// apex_shared/tests/integration/test_kafka_sink.cpp
#include <apex/shared/adapters/kafka/kafka_sink.hpp>
#include <apex/shared/adapters/kafka/kafka_producer.hpp>
#include <apex/shared/adapters/kafka/kafka_consumer.hpp>
#include <apex/shared/adapters/kafka/kafka_config.hpp>

#include <gtest/gtest.h>
#include <spdlog/spdlog.h>
#include <spdlog/logger.h>

#include <boost/asio/io_context.hpp>

#include <atomic>
#include <chrono>
#include <string>

using namespace apex::shared::adapters::kafka;

TEST(KafkaSinkIntegration, LogAppearsInKafka) {
    KafkaConfig config;
    config.brokers = "localhost:9092";
    config.log_topic = "apex-test-logs";
    config.consumer_group = "apex-sink-integration-test";
    config.consume_topics = {"apex-test-logs"};

    // KafkaSink 기반 logger 생성
    KafkaProducer producer(config);
    auto init_result = producer.init();
    ASSERT_TRUE(init_result.has_value())
        << "Producer init failed — is Kafka running?";

    auto sink = std::make_shared<KafkaSink>(producer, config.log_topic);
    auto logger = std::make_shared<spdlog::logger>("test-kafka-sink", sink);
    logger->set_level(spdlog::level::info);

    // 고유 메시지 전송
    std::string unique_marker = "sink-test-" +
        std::to_string(std::chrono::steady_clock::now().time_since_epoch().count());
    logger->info("Integration test: {}", unique_marker);
    logger->flush();
    producer.flush(std::chrono::milliseconds{5000});

    // Consumer로 수신 확인
    boost::asio::io_context io_ctx;
    KafkaConsumer consumer(config, 0, io_ctx);

    std::atomic<bool> received{false};

    consumer.set_message_callback(
        [&](std::string_view, int32_t, std::span<const uint8_t>,
            std::span<const uint8_t> payload, int64_t) {
            std::string msg(reinterpret_cast<const char*>(payload.data()), payload.size());
            if (msg.find(unique_marker) != std::string::npos) {
                received.store(true, std::memory_order_release);
            }
        });

    auto consumer_init = consumer.init();
    ASSERT_TRUE(consumer_init.has_value());
    consumer.start_consuming();

    auto deadline = std::chrono::steady_clock::now() + std::chrono::seconds{10};
    while (!received.load(std::memory_order_acquire) &&
           std::chrono::steady_clock::now() < deadline) {
        io_ctx.run_for(std::chrono::milliseconds{100});
    }

    consumer.stop_consuming();

    EXPECT_TRUE(received.load())
        << "KafkaSink log message not received in Kafka within 10 seconds";
}
```
