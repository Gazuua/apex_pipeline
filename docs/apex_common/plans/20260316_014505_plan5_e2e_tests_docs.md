```bash
#!/bin/bash
# Wait for Kafka to be ready
echo "Waiting for Kafka..."
until kafka-topics --bootstrap-server localhost:9092 --list > /dev/null 2>&1; do
    sleep 2
done
echo "Kafka is ready."

# Create topics
TOPICS=(
    "auth.requests:4:1"
    "auth.responses:4:1"
    "chat.requests:4:1"
    "chat.responses:4:1"
    "chat.messages.persist:4:1"
    "chat.messages.persist.dlq:1:1"
    "gateway.system:1:1"
)

for TOPIC_SPEC in "${TOPICS[@]}"; do
    IFS=':' read -r TOPIC PARTITIONS REPLICATION <<< "$TOPIC_SPEC"
    kafka-topics --bootstrap-server localhost:9092 \
        --create --if-not-exists \
        --topic "$TOPIC" \
        --partitions "$PARTITIONS" \
        --replication-factor "$REPLICATION"
    echo "Created topic: $TOPIC (partitions=$PARTITIONS)"
done
```

```cpp
#include <apex/auth_svc/generated/auth_messages_generated.h>
#include <apex/chat_svc/generated/chat_room_generated.h>

#include <gtest/gtest.h>

namespace apex::e2e {

class AuthE2ETest : public E2ETestFixture {};

/// 시나리오 1: 로그인 → JWT 발급 → 인증된 요청 → 응답
TEST_F(AuthE2ETest, LoginAndAuthenticatedRequest) {
    TcpClient client(io_ctx_, config_);
    client.connect();

    // 1. 로그인 요청
    auto auth = login(client, "alice", "test_password_alice");
    ASSERT_FALSE(auth.access_token.empty()) << "JWT 발급 실패";
    ASSERT_GT(auth.user_id, 0u);

    // 2. JWT 바인딩
    authenticate(client, auth.access_token);

    // 3. 인증된 요청 — 채팅방 목록 조회
    {
        flatbuffers::FlatBufferBuilder fbb(128);
        auto req = apex::chat_svc::fbs::CreateListRoomsRequest(fbb, 0, 20);
        fbb.Finish(req);
        client.send(2007, fbb.GetBufferPointer(), fbb.GetSize());
    }

    // 4. 응답 수신
    auto resp = client.recv();
    EXPECT_EQ(resp.msg_id, 2008u);  // ListRoomsResponse

    auto* list_resp = flatbuffers::GetRoot<apex::chat_svc::fbs::ListRoomsResponse>(
        resp.payload.data());
    EXPECT_EQ(list_resp->error(),
              apex::chat_svc::fbs::ChatRoomError_NONE);

    client.close();
}

/// 시나리오: JWT 없이 요청 → Gateway 에러
TEST_F(AuthE2ETest, UnauthenticatedRequestRejected) {
    TcpClient client(io_ctx_, config_);
```

```cpp
#include <apex/chat_svc/generated/chat_room_generated.h>
#include <apex/chat_svc/generated/chat_message_generated.h>

#include <gtest/gtest.h>
#include <thread>

namespace apex::e2e {

class ChatE2ETest : public E2ETestFixture {};

/// 시나리오 2: 채팅방 입장 → 메시지 전송 → 브로드캐스트 수신
TEST_F(ChatE2ETest, RoomMessageBroadcast) {
    // --- Alice 로그인 ---
    TcpClient alice(io_ctx_, config_);
    alice.connect();
    auto alice_auth = login(alice, "alice", "test_password_alice");
    authenticate(alice, alice_auth.access_token);

    // --- Bob 로그인 ---
    TcpClient bob(io_ctx_, config_);
    bob.connect();
    auto bob_auth = login(bob, "bob", "test_password_bob");
    authenticate(bob, bob_auth.access_token);

    // 1. Alice가 방 생성
    uint64_t room_id = 0;
    {
        flatbuffers::FlatBufferBuilder fbb(256);
        auto name = fbb.CreateString("E2E Test Room");
        auto req = apex::chat_svc::fbs::CreateCreateRoomRequest(fbb, name, 50);
        fbb.Finish(req);
        alice.send(2001, fbb.GetBufferPointer(), fbb.GetSize());

        auto resp = alice.recv();
        ASSERT_EQ(resp.msg_id, 2002u);
        auto* create_resp = flatbuffers::GetRoot<
            apex::chat_svc::fbs::CreateRoomResponse>(resp.payload.data());
        ASSERT_EQ(create_resp->error(), apex::chat_svc::fbs::ChatRoomError_NONE);
        room_id = create_resp->room_id();
        ASSERT_GT(room_id, 0u);
    }

    // 2. Bob이 방 입장
    {
        flatbuffers::FlatBufferBuilder fbb(128);
        auto req = apex::chat_svc::fbs::CreateJoinRoomRequest(fbb, room_id);
        fbb.Finish(req);
        bob.send(2003, fbb.GetBufferPointer(), fbb.GetSize());

        auto resp = bob.recv();
        ASSERT_EQ(resp.msg_id, 2004u);
```
