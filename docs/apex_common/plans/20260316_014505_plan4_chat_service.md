# Plan 4: Chat Service — 방 관리 + Redis Pub/Sub 브로드캐스트 + 히스토리

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan.

**Goal:** 채팅 서비스를 통한 전체 파이프라인 검증 — 방 관리, 실시간 브로드캐스트, 히스토리 저장

**Architecture:** ServiceBase 상속, Redis Pub/Sub로 브로드캐스트, Kafka→DB 히스토리 이중 경로. PostgreSQL(chat_svc 스키마).

**Tech Stack:** C++23, librdkafka, hiredis, libpq, FlatBuffers

```sql
-- Chat Service 전용 스키마
CREATE SCHEMA IF NOT EXISTS chat_svc;

-- 서비스 전용 역할 (cross-schema 접근 차단)
DO $$
BEGIN
    IF NOT EXISTS (SELECT FROM pg_roles WHERE rolname = 'chat_svc_role') THEN
        CREATE ROLE chat_svc_role LOGIN PASSWORD 'chat_svc_password';
    END IF;
END $$;

-- 권한 부여
GRANT USAGE ON SCHEMA chat_svc TO chat_svc_role;
GRANT ALL PRIVILEGES ON ALL TABLES IN SCHEMA chat_svc TO chat_svc_role;
ALTER DEFAULT PRIVILEGES IN SCHEMA chat_svc
    GRANT ALL PRIVILEGES ON TABLES TO chat_svc_role;

-- 채팅방 내 역할 타입
CREATE TYPE chat_svc.chat_role AS ENUM (
    'owner',      -- 방 생성자
    'moderator',  -- 관리자
    'member'      -- 일반 참여자
);
```

### Step 2.2: chat_rooms 테이블

**파일**: `apex_services/chat-svc/sql/002_create_tables.sql` (신규)

```sql
-- 채팅방 테이블
CREATE TABLE IF NOT EXISTS chat_svc.chat_rooms (
    room_id       BIGSERIAL PRIMARY KEY,
    room_name     VARCHAR(100)  NOT NULL,
    owner_id      BIGINT        NOT NULL,       -- users 테이블 참조 (cross-schema JOIN 금지 — 비정규화)
    max_members   INTEGER       NOT NULL DEFAULT 100,
    is_active     BOOLEAN       NOT NULL DEFAULT TRUE,
    created_at    TIMESTAMPTZ   NOT NULL DEFAULT NOW(),
    updated_at    TIMESTAMPTZ   NOT NULL DEFAULT NOW()
);

-- 인덱스: 활성 방 목록 조회 최적화
CREATE INDEX IF NOT EXISTS idx_chat_rooms_active
    ON chat_svc.chat_rooms (is_active)
    WHERE is_active = TRUE;

-- 채팅 메시지 테이블 (히스토리)
CREATE TABLE IF NOT EXISTS chat_svc.chat_messages (
    message_id    BIGSERIAL     PRIMARY KEY,
    room_id       BIGINT        NOT NULL REFERENCES chat_svc.chat_rooms(room_id),
    sender_id     BIGINT        NOT NULL,       -- 비정규화
    sender_name   VARCHAR(50)   NOT NULL,       -- 비정규화 (JOIN 방지)
    content       TEXT          NOT NULL,
    message_type  SMALLINT      NOT NULL DEFAULT 0,  -- 0=normal, 1=system, 2=whisper
    created_at    TIMESTAMPTZ   NOT NULL DEFAULT NOW()
);

-- 인덱스: 방별 최신 메시지 조회 (히스토리 페이징)
CREATE INDEX IF NOT EXISTS idx_chat_messages_room_time
    ON chat_svc.chat_messages (room_id, created_at DESC);

-- 인덱스: 특정 유저의 메시지 조회
CREATE INDEX IF NOT EXISTS idx_chat_messages_sender
    ON chat_svc.chat_messages (sender_id, created_at DESC);
```

```cpp
ChatService::on_join_room(apex::core::SessionPtr session, uint16_t msg_id,
                          const JoinRoomRequest* req)
{
    auto room_id = req->room_id();
    auto user_id = session->user_id();

    // 1. Redis에서 방 존재 여부 + 멤버 수 확인
    auto& mux = redis_data_->multiplexer(/* current core */);
    auto exists = co_await mux.command(
        fmt::format("EXISTS chat:room:{}:info", room_id));
    if (!exists || exists->as_integer() == 0) {
        co_return co_await send_room_error(session, msg_id,
            fbs::ChatRoomError::ChatRoomError_ROOM_NOT_FOUND);
    }

    // 2. 이미 참여 중인지 확인
    auto is_member = co_await mux.command(
        fmt::format("SISMEMBER chat:room:{}:members {}", room_id, user_id));
    if (is_member && is_member->as_integer() == 1) {
        co_return co_await send_room_error(session, msg_id,
            fbs::ChatRoomError::ChatRoomError_ALREADY_IN_ROOM);
    }

    // 3. 멤버 수 한도 확인
    auto member_count_reply = co_await mux.command(
        fmt::format("SCARD chat:room:{}:members", room_id));
    auto member_count = member_count_reply ? static_cast<uint32_t>(member_count_reply->as_integer()) : 0u;
    auto max_info = co_await mux.command(
        fmt::format("HGET chat:room:{}:info max_members", room_id));
    auto max_members = max_info ? static_cast<uint32_t>(max_info->as_integer()) : config_.max_room_members;

    if (member_count >= max_members) {
        co_return co_await send_room_error(session, msg_id,
            fbs::ChatRoomError::ChatRoomError_ROOM_FULL);
    }

    // 4. 멤버 추가 (Redis)
    co_await mux.pipeline({
        fmt::format("SADD chat:room:{}:members {}", room_id, user_id),
        fmt::format("SADD chat:user:{}:rooms {}", user_id, room_id),
    });

    // 5. Gateway에 Pub/Sub 구독 제어 메시지 전송
    //    → 이 세션이 pub:chat:room:{id} 채널을 구독하도록 지시
    co_await send_subscribe_control(session, fmt::format("pub:chat:room:{}", room_id));

    // 6. 응답
    auto room_name_reply = co_await mux.command(
        fmt::format("HGET chat:room:{}:info name", room_id));
    auto room_name = room_name_reply ? room_name_reply->as_string() : "";

    flatbuffers::FlatBufferBuilder fbb(256);
    auto name_off = fbb.CreateString(room_name);
    auto resp = fbs::CreateJoinRoomResponse(fbb,
        fbs::ChatRoomError::ChatRoomError_NONE, room_id, name_off, member_count + 1);
    fbb.Finish(resp);

    co_return co_await produce_response(session, 2004, fbb);
```

```cpp
boost::asio::awaitable<apex::core::Result<void>>
ChatService::produce_response(apex::core::SessionPtr session, uint32_t resp_msg_id,
                              const flatbuffers::FlatBufferBuilder& fbb)
{
    // 1. Routing Header 구성 (8바이트)
    //    direction=응답(1), delivery=단일(00)
    KafkaEnvelope envelope;
    envelope.routing.header_version = 1;
    envelope.routing.flags = RoutingFlags::RESPONSE;
    envelope.routing.msg_id = resp_msg_id;

    // 2. Metadata 구성 (32바이트)
    envelope.meta.meta_version = 1;
    envelope.meta.core_id     = session->core_id();
    envelope.meta.corr_id     = session->corr_id();
    envelope.meta.source_id   = 2;  // Chat Service = 2
    envelope.meta.session_id  = session->session_id();
    envelope.meta.timestamp   = current_timestamp_ms();

    // 3. 직렬화: [Routing 8B] + [Metadata 32B] + [Payload NB]
    auto payload = serialize_envelope(envelope, fbb);

    // 4. Kafka produce
    auto key = std::to_string(session->session_id());
    auto result = kafka_->produce(config_.kafka_response_topic, key,
        std::span<const uint8_t>(payload.data(), payload.size()));
```

```cpp
boost::asio::awaitable<apex::core::Result<void>>
ChatService::on_whisper(apex::core::SessionPtr session, uint16_t msg_id,
                        const WhisperRequest* req)
{
    auto target_user_id = req->target_user_id();
    auto content = req->content();

    // 1. 입력 검증
    if (!content || content->size() == 0) {
        co_return co_await send_message_error(session, msg_id,
            fbs::ChatMessageError::ChatMessageError_EMPTY_MESSAGE);
    }
    if (content->size() > config_.max_message_length) {
        co_return co_await send_message_error(session, msg_id,
            fbs::ChatMessageError::ChatMessageError_MESSAGE_TOO_LONG);
    }

    // 2. 대상 유저의 session_id 조회 (Redis #0 Auth 세션 or Redis #2 Chat 데이터)
    //    Redis key: session:user:{user_id} → session_id
    auto& data_mux = redis_data_->multiplexer(/* current core */);
    auto target_session = co_await data_mux.command(
        fmt::format("GET session:user:{}", target_user_id));

    if (!target_session || target_session->is_nil()) {
        co_return co_await send_message_error(session, msg_id,
            fbs::ChatMessageError::ChatMessageError_TARGET_OFFLINE);
    }

    auto target_session_id = std::stoull(target_session->as_string());
    auto timestamp = current_timestamp_ms();

    // 3. 대상에게 WhisperMessage 전송 (Kafka 응답 → Gateway → 대상 세션)
    //    Routing Header flags: 방향=응답, 전달방식=단일(00)
    //    Metadata.session_id = target_session_id
    {
        flatbuffers::FlatBufferBuilder fbb(512);
        auto sender_off  = fbb.CreateString(session->user_name());
        auto content_off = fbb.CreateString(content->str());
        auto whisper_msg = fbs::CreateWhisperMessage(fbb,
            session->user_id(), sender_off, content_off, timestamp);
        fbb.Finish(whisper_msg);

        // target session에 대한 Kafka Envelope
        KafkaEnvelope envelope;
        envelope.routing.header_version = 1;
        envelope.routing.flags = RoutingFlags::RESPONSE;  // 단일 전달
        envelope.routing.msg_id = 2023;  // WhisperMessage

        envelope.meta.meta_version = 1;
        envelope.meta.core_id     = 0;  // Gateway가 target core 결정
        envelope.meta.corr_id     = 0;  // 비요청 메시지
        envelope.meta.source_id   = 2;
        envelope.meta.session_id  = target_session_id;
        envelope.meta.timestamp   = timestamp;

        auto payload = serialize_envelope(envelope, fbb);
        kafka_->produce(config_.kafka_response_topic,
            std::to_string(target_session_id),
            std::span<const uint8_t>(payload.data(), payload.size()));
    }

    // 4. 발신자에게 확인 응답
    flatbuffers::FlatBufferBuilder fbb(128);
    auto resp = fbs::CreateWhisperResponse(fbb,
        fbs::ChatMessageError::ChatMessageError_NONE, timestamp);
    fbb.Finish(resp);

    co_return co_await produce_response(session, 2022, fbb);
}
```

```cpp
boost::asio::awaitable<apex::core::Result<void>>
ChatService::on_chat_history(apex::core::SessionPtr session, uint16_t msg_id,
                             const ChatHistoryRequest* req)
{
    auto room_id = req->room_id();
    auto user_id = session->user_id();
    auto limit   = std::min(req->limit(), static_cast<uint32_t>(config_.history_page_size));

    // 1. 방 참여 확인 (Redis) — 비참여자는 히스토리 조회 불가
    auto& data_mux = redis_data_->multiplexer(/* current core */);
    auto is_member = co_await data_mux.command(
        fmt::format("SISMEMBER chat:room:{}:members {}", room_id, user_id));
    if (!is_member || is_member->as_integer() == 0) {
        co_return co_await send_message_error(session, msg_id,
            fbs::ChatMessageError::ChatMessageError_NOT_IN_ROOM);
    }

    // 2. PostgreSQL 쿼리 — 커서 기반 페이징
    std::string sql;
    std::vector<std::string> params;

    if (req->before_message_id() > 0) {
        sql = "SELECT message_id, sender_id, sender_name, content, "
              "EXTRACT(EPOCH FROM created_at)::bigint * 1000 AS timestamp_ms "
              "FROM chat_svc.chat_messages "
              "WHERE room_id = $1 AND message_id < $2 "
              "ORDER BY created_at DESC LIMIT $3";
        params = {std::to_string(room_id),
                  std::to_string(req->before_message_id()),
                  std::to_string(limit + 1)};  // +1 for has_more
    } else {
        sql = "SELECT message_id, sender_id, sender_name, content, "
              "EXTRACT(EPOCH FROM created_at)::bigint * 1000 AS timestamp_ms "
              "FROM chat_svc.chat_messages "
              "WHERE room_id = $1 "
              "ORDER BY created_at DESC LIMIT $2";
        params = {std::to_string(room_id), std::to_string(limit + 1)};
    }

    auto result = co_await pg_->query(sql, params);
    if (!result) {
        co_return apex::core::error(result.error());
    }

    // 3. FlatBuffers 응답 빌드
    bool has_more = result->row_count() > static_cast<int>(limit);
    int display_count = has_more ? static_cast<int>(limit) : result->row_count();

    flatbuffers::FlatBufferBuilder fbb(2048);
    std::vector<flatbuffers::Offset<fbs::HistoryMessage>> msgs;

    for (int row = 0; row < display_count; ++row) {
        auto mid        = result->get<uint64_t>(row, 0);
        auto sender_id  = result->get<uint64_t>(row, 1);
        auto sender_nm  = result->get<std::string>(row, 2);
        auto content    = result->get<std::string>(row, 3);
        auto ts         = result->get<uint64_t>(row, 4);

        auto sender_off  = fbb.CreateString(sender_nm);
        auto content_off = fbb.CreateString(content);
        msgs.push_back(fbs::CreateHistoryMessage(fbb,
            mid, sender_id, sender_off, content_off, ts));
    }

    auto msgs_off = fbb.CreateVector(msgs);
    auto resp = fbs::CreateChatHistoryResponse(fbb,
        fbs::ChatMessageError::ChatMessageError_NONE, msgs_off, has_more);
    fbb.Finish(resp);

    co_return co_await produce_response(session, 2032, fbb);
}
```
