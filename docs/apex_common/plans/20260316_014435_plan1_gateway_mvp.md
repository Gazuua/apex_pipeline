# Plan 1: Gateway MVP — TLS + JWT + 라우팅 + Pub/Sub

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan.

**Goal:** WebSocket+TLS Gateway에서 JWT 검증 후 Kafka 라우팅 + 응답 전달 + Redis Pub/Sub 브로드캐스트 수신

**Architecture:** Plan 0에서 구축한 Transport concept과 Kafka Envelope 위에 Gateway 서비스를 구현. msg_id 범위 기반 라우팅, Correlation ID 요청-응답, Redis Pub/Sub 브로드캐스트.

**Tech Stack:** C++23, Boost.Asio + Beast, OpenSSL, jwt-cpp, librdkafka, hiredis

---

```cpp
#pragma once

#include <chrono>
#include <cstdint>
#include <string>
#include <vector>

namespace apex::gateway {

struct TlsConfig {
    std::string cert_file;            // PEM 인증서 경로
    std::string key_file;             // PEM 개인키 경로
    std::string ca_file;              // CA 인증서 (클라이언트 검증용, 선택)
    bool require_client_cert = false; // mTLS (향후)
};

struct JwtConfig {
    std::string secret;               // HMAC 비밀키 (HS256)
    std::string public_key_file;      // RSA/EC 공개키 파일 (RS256/ES256, 향후)
    std::string algorithm = "HS256";
    std::chrono::seconds clock_skew{30};     // 시간 허용 오차
    std::vector<uint32_t> sensitive_msg_ids; // Redis 블랙리스트 조회 대상
};

struct RouteEntry {
    uint32_t range_begin;             // msg_id 범위 시작 (포함)
    uint32_t range_end;               // msg_id 범위 끝 (포함)
    std::string kafka_topic;          // 대상 Kafka 토픽
};

struct GatewayConfig {
    // Network
    uint16_t ws_port = 8443;          // WebSocket + TLS 포트
    uint32_t num_cores = 1;

    // TLS
    TlsConfig tls;

    // JWT
    JwtConfig jwt;

    // Routing
    std::vector<RouteEntry> routes;   // msg_id 범위 → Kafka 토픽 매핑
    uint32_t system_range_end = 999;  // 시스템 msg_id 범위 [0, 999]

    // Kafka
    std::string kafka_brokers = "localhost:9092";
    std::string kafka_consumer_group = "gateway";
    std::string kafka_response_topic = "gateway.responses";

    // Redis (Pub/Sub)
    std::string redis_pubsub_host = "localhost";
    uint16_t redis_pubsub_port = 6379;
    std::string redis_pubsub_password;
    std::vector<std::string> global_channels = {"pub:global:chat"};

    // Redis (Auth / 블랙리스트 조회)
    std::string redis_auth_host = "localhost";
    uint16_t redis_auth_port = 6379;
    std::string redis_auth_password;

    // Timeouts
    std::chrono::milliseconds request_timeout{5000};  // Pending request 타임아웃
    size_t max_pending_per_core = 65536;               // per-core pending map 최대 크기
};

} // namespace apex::gateway
```

```cpp
#include <apex/gateway/gateway_config_parser.hpp>

#include <toml++/toml.hpp>
#include <spdlog/spdlog.h>

namespace apex::gateway {

apex::core::Result<GatewayConfig>
parse_gateway_config(std::string_view path) {
    try {
        auto tbl = toml::parse_file(path);
        GatewayConfig cfg;

        // [server]
        if (auto server = tbl["server"].as_table()) {
            cfg.ws_port = server->get("ws_port")
                ->value_or(static_cast<int64_t>(8443));
            cfg.num_cores = server->get("num_cores")
                ->value_or(static_cast<int64_t>(1));
        }

        // [tls]
        if (auto tls = tbl["tls"].as_table()) {
            cfg.tls.cert_file = tls->get("cert_file")
                ->value_or(std::string{});
            cfg.tls.key_file = tls->get("key_file")
                ->value_or(std::string{});
            cfg.tls.ca_file = tls->get("ca_file")
                ->value_or(std::string{});
        }

        // [jwt]
        if (auto jwt = tbl["jwt"].as_table()) {
            cfg.jwt.secret = jwt->get("secret")
                ->value_or(std::string{});
            cfg.jwt.algorithm = jwt->get("algorithm")
                ->value_or(std::string{"HS256"});
            cfg.jwt.clock_skew = std::chrono::seconds{
                jwt->get("clock_skew_seconds")->value_or(int64_t{30})};

            if (auto arr = jwt->get("sensitive_msg_ids")->as_array()) {
                for (auto& elem : *arr) {
                    cfg.jwt.sensitive_msg_ids.push_back(
                        static_cast<uint32_t>(elem.value_or(int64_t{0})));
                }
            }
        }

        // [[routes]]
        if (auto routes = tbl["routes"].as_array()) {
            for (auto& entry : *routes) {
                if (auto t = entry.as_table()) {
                    RouteEntry r;
                    r.range_begin = static_cast<uint32_t>(
                        t->get("range_begin")->value_or(int64_t{0}));
                    r.range_end = static_cast<uint32_t>(
                        t->get("range_end")->value_or(int64_t{0}));
                    r.kafka_topic = t->get("kafka_topic")
                        ->value_or(std::string{});
                    cfg.routes.push_back(std::move(r));
                }
            }
        }

        // [kafka]
        if (auto kafka = tbl["kafka"].as_table()) {
            cfg.kafka_brokers = kafka->get("brokers")
                ->value_or(std::string{"localhost:9092"});
            cfg.kafka_consumer_group = kafka->get("consumer_group")
                ->value_or(std::string{"gateway"});
            cfg.kafka_response_topic = kafka->get("response_topic")
                ->value_or(std::string{"gateway.responses"});
        }

        // [redis.pubsub]
        if (auto rpub = tbl["redis"]["pubsub"].as_table()) {
            cfg.redis_pubsub_host = rpub->get("host")
                ->value_or(std::string{"localhost"});
            cfg.redis_pubsub_port = static_cast<uint16_t>(
                rpub->get("port")->value_or(int64_t{6379}));
            cfg.redis_pubsub_password = rpub->get("password")
                ->value_or(std::string{});
        }

        // [redis.auth]
        if (auto rauth = tbl["redis"]["auth"].as_table()) {
            cfg.redis_auth_host = rauth->get("host")
                ->value_or(std::string{"localhost"});
            cfg.redis_auth_port = static_cast<uint16_t>(
                rauth->get("port")->value_or(int64_t{6379}));
            cfg.redis_auth_password = rauth->get("password")
                ->value_or(std::string{});
        }

        return cfg;

    } catch (const toml::parse_error& e) {
        spdlog::error("Gateway config parse error: {}", e.what());
        return apex::core::error(apex::core::ErrorCode::ConfigParseFailed);
    }
}

} // namespace apex::gateway
```

```cpp
    std::shared_ptr<spdlog::logger> logger_;
```

```cpp
#include <apex/gateway/gateway_config.hpp>
#include <apex/gateway/gateway_config_parser.hpp>
#include <apex/gateway/gateway_service.hpp>

#include <apex/core/server.hpp>
#include <apex/shared/protocols/websocket/websocket_protocol.hpp>
#include <apex/shared/adapters/adapter_base.hpp>
#include <apex/shared/adapters/kafka/kafka_adapter.hpp>
#include <apex/shared/adapters/redis/redis_adapter.hpp>

#include <spdlog/spdlog.h>

#include <cstdlib>

int main(int argc, char* argv[]) {
    spdlog::set_level(spdlog::level::info);
    spdlog::info("Apex Gateway starting...");

    // Config 경로 (기본: gateway.toml)
    std::string config_path = "gateway.toml";
    if (argc > 1) {
        config_path = argv[1];
    }

    // Config 파싱
    auto config_result = apex::gateway::parse_gateway_config(config_path);
    if (!config_result) {
        spdlog::error("Failed to parse config: {}", config_path);
        return EXIT_FAILURE;
    }
    auto& gw_config = *config_result;

    // Kafka 설정
    apex::shared::adapters::kafka::KafkaConfig kafka_cfg;
    kafka_cfg.brokers = gw_config.kafka_brokers;
    kafka_cfg.consumer_group = gw_config.kafka_consumer_group;
    kafka_cfg.consume_topics = {gw_config.kafka_response_topic};

    // Redis 설정 (Auth 전용)
    apex::shared::adapters::redis::RedisConfig redis_auth_cfg;
    redis_auth_cfg.host = gw_config.redis_auth_host;
    redis_auth_cfg.port = gw_config.redis_auth_port;
    redis_auth_cfg.password = gw_config.redis_auth_password;

    // Server 구성
    apex::core::Server server({
        .num_cores = gw_config.num_cores,
        .heartbeat_timeout_ticks = 300,
    });

    server
        .listen<apex::shared::protocols::websocket::WebSocketProtocol>(
            gw_config.ws_port)
        .add_adapter<apex::shared::adapters::kafka::KafkaAdapter>(kafka_cfg)
        .add_adapter<apex::shared::adapters::redis::RedisAdapter>(redis_auth_cfg)
        .add_service<apex::gateway::GatewayService>(gw_config)
        .run();

    spdlog::info("Apex Gateway stopped.");
    return EXIT_SUCCESS;
}
```

### Step 2.2: TlsTcpTransport 실제 구현

**파일**: `apex_shared/lib/protocols/tcp/include/apex/shared/protocols/tcp/tls_tcp_transport.hpp`

- [ ] Plan 0의 스켈레톤을 실제 구현으로 교체:

```cpp
#pragma once

#include <apex/core/transport.hpp>

#include <boost/asio/as_tuple.hpp>
#include <boost/asio/ip/tcp.hpp>
#include <boost/asio/ssl.hpp>
#include <boost/asio/use_awaitable.hpp>

#include <spdlog/spdlog.h>

#include <string>

namespace apex::shared::protocols::tcp {

/// TLS TCP transport — OpenSSL 기반 TLS 연결.
/// per-core SSL_CTX 지원 (Seastar 방식, lock contention 제거).
struct TlsTcpTransport {
    struct Config {
        std::string cert_file;
        std::string key_file;
        std::string ca_file;
        bool require_client_cert = false;
    };

    using Socket = boost::asio::ssl::stream<boost::asio::ip::tcp::socket>;

    static Socket make_socket(boost::asio::io_context& ctx) {
        // 기본 SSL context 생성 — 실제로는 per-core context를 외부에서 주입받아야 함.
        // make_socket은 Transport concept 만족용이며, 실제 사용 시
        // make_socket_with_context를 호출한다.
        //
        // NOTE: 이 기본 context는 테스트/concept 검증용.
        // 실제 경로에서는 Listener가 per-core ssl::context를 소유하고
        // make_socket_with_context로 소켓을 생성한다.
        static boost::asio::ssl::context default_ctx(
            boost::asio::ssl::context::tlsv13);
        return Socket(ctx, default_ctx);
    }

    /// per-core SSL context를 받아 소켓 생성 (실제 사용 경로).
    static Socket make_socket_with_context(
            boost::asio::io_context& ctx,
            boost::asio::ssl::context& ssl_ctx) {
        return Socket(ctx, ssl_ctx);
    }

    /// SSL context 초기화 (per-core).
    /// @return 초기화된 ssl::context. 호출자가 수명 관리.
    [[nodiscard]] static boost::asio::ssl::context
    create_ssl_context(const Config& cfg) {
        boost::asio::ssl::context ctx(boost::asio::ssl::context::tlsv13);

        ctx.set_options(
            boost::asio::ssl::context::default_workarounds |
            boost::asio::ssl::context::no_sslv2 |
            boost::asio::ssl::context::no_sslv3 |
            boost::asio::ssl::context::no_tlsv1 |
            boost::asio::ssl::context::no_tlsv1_1 |
            boost::asio::ssl::context::single_dh_use);

        ctx.use_certificate_chain_file(cfg.cert_file);
        ctx.use_private_key_file(cfg.key_file,
                                  boost::asio::ssl::context::pem);

        if (!cfg.ca_file.empty()) {
            ctx.load_verify_file(cfg.ca_file);
        }

        if (cfg.require_client_cert) {
            ctx.set_verify_mode(boost::asio::ssl::verify_peer |
                                boost::asio::ssl::verify_fail_if_no_peer_cert);
        }

        return ctx;
    }

    [[nodiscard]] static boost::asio::awaitable<apex::core::Result<void>>
    async_accept(boost::asio::ip::tcp::acceptor& acceptor, Socket& sock) {
        auto [ec] = co_await acceptor.async_accept(
            sock.lowest_layer(),
            boost::asio::as_tuple(boost::asio::use_awaitable));
        if (ec) {
            spdlog::warn("TLS accept failed: {}", ec.message());
            co_return apex::core::error(apex::core::ErrorCode::AcceptFailed);
        }
        co_return apex::core::ok();
    }

    [[nodiscard]] static boost::asio::awaitable<apex::core::Result<void>>
    async_handshake(Socket& sock, const Config&) {
        auto [ec] = co_await sock.async_handshake(
            boost::asio::ssl::stream_base::server,
            boost::asio::as_tuple(boost::asio::use_awaitable));
        if (ec) {
            spdlog::warn("TLS handshake failed: {}", ec.message());
            co_return apex::core::error(apex::core::ErrorCode::HandshakeFailed);
        }
        co_return apex::core::ok();
    }

    static boost::asio::awaitable<void>
    async_shutdown(Socket& sock) {
        boost::system::error_code ec;
        // SSL shutdown은 양방향이므로 타임아웃 처리 필요할 수 있음.
        // 에러 무시 — 클라이언트가 이미 끊었을 수 있다.
        co_await sock.async_shutdown(
```

### Step 2.6: Session 소켓 타입 추상화

현재 `Session`은 `boost::asio::ip::tcp::socket`을 하드코딩한다. TLS 지원을 위해 `ssl::stream<tcp::socket>`도 지원해야 한다.

**전략: 타입 소거(type-erasure) 래퍼**

Session을 템플릿화하면 SessionManager, SessionPtr 등 광범위한 변경이 필요하다. 대신 Boost.Beast의 `tcp_stream` / `ssl_stream`처럼 소켓을 variant 또는 type-erased wrapper로 감싼다.

```cpp
// apex_core/include/apex/core/transport_socket.hpp (신규)
#pragma once

#include <boost/asio/ip/tcp.hpp>
#include <boost/asio/awaitable.hpp>

#include <cstddef>
#include <memory>
#include <span>
#include <variant>

namespace apex::core {

/// Transport 소켓 래퍼 — TLS/Plain 통합.
/// Session이 소켓 타입을 알 필요 없이 읽기/쓰기를 수행할 수 있게 한다.
///
/// 전략: std::variant<tcp::socket, ssl::stream<tcp::socket>>
/// 비용: 분기 1회 (hot path에서 branch predictor가 학습)
///
/// 향후 필요 시 virtual 기반 type-erasure로 전환 가능하지만,
/// variant의 성능이 충분하고 코드가 더 간결하다.
class TransportSocket {
public:
    using TcpSocket = boost::asio::ip::tcp::socket;

    explicit TransportSocket(TcpSocket socket);

    // TLS 소켓 지원은 OpenSSL 통합 시 추가
    // explicit TransportSocket(boost::asio::ssl::stream<TcpSocket> socket);

    /// 하위 tcp::socket 참조 (TCP nodelay, remote_endpoint 등)
    [[nodiscard]] TcpSocket& tcp_layer() noexcept;
    [[nodiscard]] const TcpSocket& tcp_layer() const noexcept;

    /// 소켓 닫기
    void close(boost::system::error_code& ec) noexcept;

    /// 소켓 유효 여부
    [[nodiscard]] bool is_open() const noexcept;

private:
    TcpSocket socket_;
    // 향후: std::variant<TcpSocket, SslSocket> socket_;
};

} // namespace apex::core
```

> **Note**: Session의 소켓 타입을 TransportSocket으로 교체하는 것은 영향 범위가 크다. 이 단계에서는 TransportSocket 정의만 하고, 실제 Session 통합은 Listener<P,T>에서 accept 후 Session을 생성할 때 TransportSocket을 주입하는 방식으로 점진적으로 진행한다. 전체 전환은 별도 리팩토링 태스크로 분리 가능.

```cpp
#include <spdlog/spdlog.h>

#include <algorithm>

namespace apex::gateway {

JwtVerifier::JwtVerifier(const JwtConfig& config)
    : config_(config)
    , verifier_(jwt::verify()
        .allow_algorithm(jwt::algorithm::hs256{config.secret})
        .with_issuer("apex")
        .leeway(static_cast<uint64_t>(config.clock_skew.count()))) {
}

apex::core::Result<JwtClaims>
JwtVerifier::verify(std::string_view token) const {
    try {
        auto decoded = jwt::decode(std::string(token));
        verifier_.verify(decoded);

        JwtClaims claims;
        claims.user_id = decoded.get_payload_claim("uid")
            .as_integer();
        claims.username = decoded.get_payload_claim("sub")
            .as_string();
        claims.expires_at = decoded.get_expires_at();

        return claims;

    } catch (const jwt::error::token_verification_exception& e) {
        spdlog::debug("JWT verification failed: {}", e.what());
        return apex::core::error(apex::core::ErrorCode::JwtVerifyFailed);
    } catch (const jwt::error::claim_not_present_exception& e) {
        spdlog::debug("JWT missing claim: {}", e.what());
        return apex::core::error(apex::core::ErrorCode::JwtVerifyFailed);
    } catch (const std::exception& e) {
        spdlog::warn("JWT unexpected error: {}", e.what());
        return apex::core::error(apex::core::ErrorCode::JwtVerifyFailed);
    }
}

bool JwtVerifier::is_sensitive(uint32_t msg_id) const noexcept {
    // 선형 검색 — sensitive_msg_ids는 보통 10개 이하
    return std::ranges::find(config_.sensitive_msg_ids, msg_id)
        != config_.sensitive_msg_ids.end();
}
```

```cpp
#pragma once

#include <apex/gateway/gateway_config.hpp>
#include <apex/core/result.hpp>

#include <algorithm>
#include <cstdint>
#include <memory>
#include <optional>
#include <string>
#include <string_view>
#include <vector>

namespace apex::gateway {

/// msg_id 범위 기반 라우팅 테이블.
/// 이진 탐색으로 O(log N) 조회.
/// 원자적 교체를 위해 shared_ptr로 래핑.
class RouteTable {
public:
    struct Entry {
        uint32_t range_begin;
        uint32_t range_end;      // 포함 (inclusive)
        std::string kafka_topic;

        // upper_bound 비교용: range_end 기준 정렬
        bool operator<(uint32_t msg_id) const noexcept {
            return range_end < msg_id;
        }
    };

    /// 라우팅 엔트리 목록으로 테이블 생성.
    /// 범위 겹침/빈틈을 검증한다.
    /// @return 유효한 RouteTable 또는 에러
    [[nodiscard]] static apex::core::Result<RouteTable>
    build(std::vector<RouteEntry> routes);

    /// msg_id에 대한 Kafka 토픽 조회.
    /// @return 토픽 이름 또는 nullopt (라우팅 규칙 없음)
    [[nodiscard]] std::optional<std::string_view>
    resolve(uint32_t msg_id) const noexcept;

    /// 엔트리 수.
    [[nodiscard]] size_t size() const noexcept { return entries_.size(); }

    /// 유효성 검증 — 범위 겹침/빈틈 확인.
    [[nodiscard]] apex::core::Result<void> validate() const;

private:
    std::vector<Entry> entries_;  // range_end 기준 정렬
};

/// 원자적 교체를 위한 shared_ptr 래퍼.
/// Hot-reload 시 새 RouteTable을 생성 후 atomic store로 교체.
using RouteTablePtr = std::shared_ptr<const RouteTable>;

} // namespace apex::gateway
```

### Step 4.2: RouteTable 구현

**파일 (신규)**: `apex_services/gateway/src/route_table.cpp`

- [ ] 구현:

```cpp
#include <apex/gateway/route_table.hpp>

#include <spdlog/spdlog.h>

#include <algorithm>

namespace apex::gateway {

apex::core::Result<RouteTable>
RouteTable::build(std::vector<RouteEntry> routes) {
    RouteTable table;

    for (auto& r : routes) {
        if (r.range_begin > r.range_end) {
            spdlog::error("Invalid route: begin({}) > end({})",
                r.range_begin, r.range_end);
            return apex::core::error(apex::core::ErrorCode::ConfigParseFailed);
        }
        table.entries_.push_back(Entry{
            .range_begin = r.range_begin,
            .range_end = r.range_end,
            .kafka_topic = std::move(r.kafka_topic),
        });
    }

    // range_end 기준 정렬 (upper_bound 사용을 위해)
    std::ranges::sort(table.entries_, [](const Entry& a, const Entry& b) {
        return a.range_end < b.range_end;
    });

    // 유효성 검증
    auto result = table.validate();
    if (!result) return apex::core::error(result.error());

    return table;
}

std::optional<std::string_view>
RouteTable::resolve(uint32_t msg_id) const noexcept {
    // upper_bound: range_end >= msg_id인 첫 번째 엔트리 찾기
    auto it = std::upper_bound(entries_.begin(), entries_.end(), msg_id,
        [](uint32_t id, const Entry& entry) {
            return id < entry.range_begin;
            // 아니다 — upper_bound의 비교는 (value, element) 순서
        });

    // 올바른 이진 탐색: range_end >= msg_id인 첫 엔트리
    // std::lower_bound 사용이 더 적절
    auto it2 = std::ranges::lower_bound(entries_, msg_id,
        std::less<>{}, &Entry::range_end);

    if (it2 != entries_.end() &&
        msg_id >= it2->range_begin && msg_id <= it2->range_end) {
        return it2->kafka_topic;
    }
    return std::nullopt;
}

apex::core::Result<void> RouteTable::validate() const {
    for (size_t i = 1; i < entries_.size(); ++i) {
        // 범위 겹침 검사: 이전 엔트리의 range_end >= 현재의 range_begin
        if (entries_[i - 1].range_end >= entries_[i].range_begin) {
            spdlog::error("Route overlap: [{}, {}] and [{}, {}]",
                entries_[i - 1].range_begin, entries_[i - 1].range_end,
                entries_[i].range_begin, entries_[i].range_end);
            return apex::core::error(apex::core::ErrorCode::ConfigParseFailed);
        }
    }
    return apex::core::ok();
}

} // namespace apex::gateway
```

```cpp
#include <apex/gateway/message_router.hpp>

#include <spdlog/spdlog.h>

#include <chrono>

namespace apex::gateway {

using namespace apex::shared::protocols::kafka;

MessageRouter::MessageRouter(
    apex::shared::adapters::kafka::KafkaAdapter& kafka,
    RouteTablePtr initial_table,
    uint32_t core_id)
    : kafka_(kafka)
    , route_table_(std::move(initial_table))
    , core_id_(core_id) {}

apex::core::Result<void>
MessageRouter::route(apex::core::SessionPtr session,
                     const apex::core::WireHeader& header,
                     std::span<const uint8_t> payload,
                     uint64_t user_id,
                     uint64_t corr_id) {
    // 1. 라우팅 테이블에서 토픽 조회
    auto table = route_table_.load(std::memory_order_acquire);
    auto topic = table->resolve(header.msg_id);
    if (!topic) {
        spdlog::warn("No route for msg_id: {}", header.msg_id);
        return apex::core::error(apex::core::ErrorCode::RouteNotFound);
    }

    // 2. Kafka Envelope 직렬화
    auto envelope = build_envelope(
        header, payload, session->id(), corr_id, core_id_);

    // 3. Kafka produce (session_id를 키로 사용 — 파티션 균등 분산)
    auto session_key = std::to_string(session->id());
    return kafka_.produce(*topic, session_key,
        std::span<const uint8_t>(envelope));
}

void MessageRouter::update_table(RouteTablePtr new_table) {
    route_table_.store(std::move(new_table), std::memory_order_release);
}

RouteTablePtr MessageRouter::current_table() const {
    return route_table_.load(std::memory_order_acquire);
}

std::vector<uint8_t>
MessageRouter::build_envelope(
    const apex::core::WireHeader& header,
    std::span<const uint8_t> payload,
    uint64_t session_id,
    uint64_t corr_id,
    uint16_t core_id) {

    // Routing Header (8B) + Metadata (32B) + Payload
    std::vector<uint8_t> buf(ENVELOPE_HEADER_SIZE + payload.size());

    // Routing Header
    RoutingHeader rh;
    rh.msg_id = header.msg_id;
    rh.flags = 0;  // 요청, 단일 전달, 비압축
    rh.serialize(std::span<uint8_t, RoutingHeader::SIZE>(
        buf.data(), RoutingHeader::SIZE));

    // Metadata Prefix
    MetadataPrefix meta;
    meta.core_id = core_id;
    meta.corr_id = corr_id;
    meta.source_id = source_ids::GATEWAY;
    meta.session_id = session_id;
    meta.timestamp = static_cast<uint64_t>(
        std::chrono::duration_cast<std::chrono::milliseconds>(
            std::chrono::system_clock::now().time_since_epoch())
        .count());
    meta.serialize(std::span<uint8_t, MetadataPrefix::SIZE>(
        buf.data() + RoutingHeader::SIZE, MetadataPrefix::SIZE));

    // Payload 복사
    if (!payload.empty()) {
        std::memcpy(buf.data() + ENVELOPE_HEADER_SIZE,
                    payload.data(), payload.size());
    }

    return buf;
}

uint64_t MessageRouter::generate_corr_id() noexcept {
    // per-core이므로 atomic 불필요
    // 상위 16비트: core_id, 하위 48비트: monotonic counter
    return (static_cast<uint64_t>(core_id_) << 48) | (++corr_counter_);
}
```

```cpp
#include <spdlog/spdlog.h>

namespace apex::gateway {

FileWatcher::FileWatcher(boost::asio::io_context& io_ctx,
                         std::string path,
                         Callback callback,
                         std::chrono::milliseconds interval)
    : io_ctx_(io_ctx)
    , path_(std::move(path))
    , callback_(std::move(callback))
    , interval_(interval)
    , timer_(std::make_unique<boost::asio::steady_timer>(io_ctx)) {
    // 초기 수정 시간 기록
    std::error_code ec;
    last_write_time_ = std::filesystem::last_write_time(path_, ec);
    if (ec) {
        spdlog::warn("FileWatcher: cannot stat {}: {}", path_, ec.message());
    }
}

FileWatcher::~FileWatcher() {
    stop();
}

void FileWatcher::start() {
    running_ = true;
    schedule_check();
    spdlog::info("FileWatcher started: {}", path_);
}

void FileWatcher::stop() {
    running_ = false;
    timer_->cancel();
}

void FileWatcher::schedule_check() {
    if (!running_) return;
    timer_->expires_after(interval_);
    timer_->async_wait([this](boost::system::error_code ec) {
        if (ec || !running_) return;
        check_file();
        schedule_check();
    });
}

void FileWatcher::check_file() {
    std::error_code ec;
    auto current = std::filesystem::last_write_time(path_, ec);
    if (ec) {
        spdlog::debug("FileWatcher: stat failed for {}: {}",
            path_, ec.message());
        return;
    }

    if (current != last_write_time_) {
        last_write_time_ = current;
        spdlog::info("FileWatcher: change detected in {}", path_);
        if (callback_) {
            callback_(path_);
        }
    }
}
```

### Step 6.2: PendingRequestsMap 구현

**파일 (신규)**: `apex_services/gateway/src/pending_requests.cpp`

- [ ] 구현:

```cpp
#include <apex/gateway/pending_requests.hpp>

#include <spdlog/spdlog.h>

namespace apex::gateway {

PendingRequestsMap::PendingRequestsMap(
    size_t max_entries,
    std::chrono::milliseconds timeout)
    : max_entries_(max_entries)
    , timeout_(timeout) {}

apex::core::Result<void>
PendingRequestsMap::insert(uint64_t corr_id,
                           apex::core::SessionId session_id,
                           uint32_t original_msg_id) {
    if (map_.size() >= max_entries_) {
        spdlog::warn("PendingRequestsMap full ({}/{})",
            map_.size(), max_entries_);
        return apex::core::error(apex::core::ErrorCode::PendingMapFull);
    }

    map_.emplace(corr_id, PendingEntry{
        .session_id = session_id,
        .original_msg_id = original_msg_id,
        .deadline = std::chrono::steady_clock::now() + timeout_,
    });

    return apex::core::ok();
}

std::optional<PendingRequestsMap::PendingEntry>
PendingRequestsMap::extract(uint64_t corr_id) {
    auto it = map_.find(corr_id);
    if (it == map_.end()) return std::nullopt;

    auto entry = std::move(it->second);
    map_.erase(it);
    return entry;
}

void PendingRequestsMap::sweep_expired(
    std::function<void(uint64_t, const PendingEntry&)> callback) {
    auto now = std::chrono::steady_clock::now();
    for (auto it = map_.begin(); it != map_.end();) {
        if (it->second.deadline <= now) {
            if (callback) callback(it->first, it->second);
            it = map_.erase(it);
        } else {
            ++it;
        }
    }
}
```

```cpp
#include <apex/core/cross_core_call.hpp>
#include <spdlog/spdlog.h>

namespace apex::gateway {

using namespace apex::shared::protocols::kafka;

ResponseDispatcher::ResponseDispatcher(
    apex::core::CoreEngine& engine,
    std::vector<PendingRequestsMap*> pending_maps,
    std::vector<apex::core::SessionManager*> session_mgrs)
    : engine_(engine)
    , pending_maps_(std::move(pending_maps))
    , session_mgrs_(std::move(session_mgrs)) {}

apex::core::Result<void>
ResponseDispatcher::on_response(std::span<const uint8_t> payload) {
    // 1. Routing Header 파싱
    if (payload.size() < ENVELOPE_HEADER_SIZE) {
        spdlog::warn("Response too short: {} bytes", payload.size());
        return apex::core::error(apex::core::ErrorCode::ParseFailed);
    }

    auto rh_result = RoutingHeader::parse(payload.subspan(0, RoutingHeader::SIZE));
    if (!rh_result) {
        spdlog::warn("Invalid routing header in response");
        return apex::core::error(apex::core::ErrorCode::ParseFailed);
    }

    // 2. Metadata 파싱
    auto meta_result = MetadataPrefix::parse(
        payload.subspan(RoutingHeader::SIZE, MetadataPrefix::SIZE));
    if (!meta_result) {
        spdlog::warn("Invalid metadata in response");
        return apex::core::error(apex::core::ErrorCode::ParseFailed);
    }

    auto& routing = *rh_result;
    auto& meta = *meta_result;

    // 3. 대상 코어의 PendingMap에서 corr_id 조회
    uint16_t target_core = meta.core_id;
    if (target_core >= pending_maps_.size()) {
        spdlog::warn("Invalid core_id {} in response", target_core);
        return apex::core::error(apex::core::ErrorCode::ParseFailed);
    }

    auto pending = pending_maps_[target_core]->extract(meta.corr_id);
    if (!pending) {
        // 이미 타임아웃되었거나 중복 응답
        spdlog::debug("No pending request for corr_id: {}", meta.corr_id);
        return apex::core::ok();  // 무시
    }

    // 4. FlatBuffers payload 추출
    auto fbs_payload = payload.subspan(ENVELOPE_HEADER_SIZE);

    // 5. WireHeader 응답 빌드
    auto wire_response = build_wire_response(
        routing, fbs_payload, pending->original_msg_id);

    // 6. 대상 코어의 세션에 응답 전달
    auto session_id = pending->session_id;
    auto* session_mgr = session_mgrs_[target_core];

    // cross_core_post: fire-and-forget로 대상 코어에서 실행
    return apex::core::cross_core_post(
        engine_, target_core,
        [session_mgr, session_id,
         response = std::move(wire_response)]() {
            auto session = session_mgr->find(session_id);
            if (!session || !session->is_open()) return;
            session->enqueue_write(
                std::vector<uint8_t>(response.begin(), response.end()));
        });
}

std::vector<uint8_t>
ResponseDispatcher::build_wire_response(
    const RoutingHeader& routing,
    std::span<const uint8_t> fbs_payload,
    uint32_t original_msg_id) {

    apex::core::WireHeader wh;
    wh.msg_id = original_msg_id;
    wh.body_size = static_cast<uint32_t>(fbs_payload.size());
    wh.flags = 0;

    // Routing Header의 에러 비트를 WireHeader flags에 반영
    if (routing.flags & routing_flags::ERROR_BIT) {
        wh.flags |= apex::core::wire_flags::ERROR_RESPONSE;
    }

    std::vector<uint8_t> buf(apex::core::WireHeader::SIZE + fbs_payload.size());
    std::array<uint8_t, apex::core::WireHeader::SIZE> hdr_buf{};
    wh.serialize(hdr_buf);
    std::memcpy(buf.data(), hdr_buf.data(), hdr_buf.size());

    if (!fbs_payload.empty()) {
        std::memcpy(buf.data() + apex::core::WireHeader::SIZE,
                    fbs_payload.data(), fbs_payload.size());
    }

    return buf;
}
```

```cpp
void PubSubListener::start() {

    spdlog::info("PubSubListener started ({}:{})",
        config_.host, config_.port);

void PubSubListener::stop() {
    running_ = false;
    if (ctx_) {
        redisAsyncDisconnect(ctx_);
    }
    if (thread_.joinable()) {
        thread_.join();
    }
    spdlog::info("PubSubListener stopped");

void PubSubListener::subscribe(const std::string& channel) {
    std::lock_guard lock(channels_mutex_);
    subscribed_channels_.insert(channel);
    if (ctx_) {
        redisAsyncCommand(ctx_, nullptr, nullptr,
            "SUBSCRIBE %s", channel.c_str());
    }
}

void PubSubListener::unsubscribe(const std::string& channel) {
    std::lock_guard lock(channels_mutex_);
    subscribed_channels_.erase(channel);
    if (ctx_) {
        redisAsyncCommand(ctx_, nullptr, nullptr,
            "UNSUBSCRIBE %s", channel.c_str());
    }
}

void PubSubListener::run_thread() {
    while (running_) {
        // 동기 연결로 Pub/Sub 루프 실행.
        // hiredis async + libevent/libev 또는 수동 fd 폴링.
        //
        // 간단한 구현: 동기 hiredis로 blocking subscribe.
        redisContext* sync_ctx = redisConnect(
            config_.host.c_str(), config_.port);
        if (!sync_ctx || sync_ctx->err) {
            spdlog::error("PubSub connect failed: {}",
                sync_ctx ? sync_ctx->errstr : "null context");
            if (sync_ctx) redisFree(sync_ctx);
            std::this_thread::sleep_for(std::chrono::seconds{1});
            continue;
        }

        // AUTH
        if (!config_.password.empty()) {
            auto* reply = static_cast<redisReply*>(
                redisCommand(sync_ctx, "AUTH %s", config_.password.c_str()));
            if (reply) freeReplyObject(reply);
        }

        // 초기 구독
        {
            std::lock_guard lock(channels_mutex_);
            for (const auto& ch : config_.initial_channels) {
                subscribed_channels_.insert(ch);
            }
            for (const auto& ch : subscribed_channels_) {
                auto* reply = static_cast<redisReply*>(
                    redisCommand(sync_ctx, "SUBSCRIBE %s", ch.c_str()));
                if (reply) freeReplyObject(reply);
            }
        }

        // 메시지 수신 루프
        while (running_) {
            redisReply* reply = nullptr;
            int status = redisGetReply(sync_ctx,
                reinterpret_cast<void**>(&reply));
            if (status != REDIS_OK || !reply) {
                spdlog::warn("PubSub read error, reconnecting...");
                break;
            }

            if (reply->type == REDIS_REPLY_ARRAY &&
                reply->elements == 3 &&
                std::string_view(reply->element[0]->str) == "message") {

                std::string_view channel(
                    reply->element[1]->str, reply->element[1]->len);
                auto* data = reinterpret_cast<const uint8_t*>(
                    reply->element[2]->str);
                size_t len = reply->element[2]->len;

                if (on_message_) {
                    on_message_(channel,
                        std::span<const uint8_t>(data, len));
                }
            }

            freeReplyObject(reply);
        }

        redisFree(sync_ctx);

        if (running_) {
            spdlog::info("PubSub reconnecting in 1 second...");
            std::this_thread::sleep_for(std::chrono::seconds{1});
        }
    }
}
```

```cpp
#include <apex/core/cross_core_call.hpp>
#include <apex/core/wire_header.hpp>
#include <spdlog/spdlog.h>

namespace apex::gateway {

BroadcastFanout::BroadcastFanout(
    apex::core::CoreEngine& engine,
    const ChannelSessionMap& channel_map,
    std::vector<apex::core::SessionManager*> session_mgrs)
    : engine_(engine)
    , channel_map_(channel_map)
    , session_mgrs_(std::move(session_mgrs)) {}

void BroadcastFanout::fanout(std::string_view channel,
                             std::span<const uint8_t> message) {
    auto groups = channel_map_.get_subscribers(std::string(channel));
    if (groups.empty()) return;

    // 브로드캐스트 msg_id는 서비스가 Envelope payload에 포함.
    // Gateway는 opaque하게 전달 — WireHeader 없이 raw 데이터를 전송.
    // 또는 서비스가 WireHeader까지 포함한 완전한 프레임을 제공.
    //
    // 설계 결정: 서비스 → Redis PUBLISH 시 WireHeader + payload 전체를
    // 직렬화해서 publish. Gateway는 그대로 세션에 전달.
    auto data = std::vector<uint8_t>(message.begin(), message.end());

    for (const auto& group : groups) {
        auto shared_data = std::make_shared<std::vector<uint8_t>>(data);
        auto session_ids = group.session_ids;
        auto core_id = group.core_id;
        auto* mgr = session_mgrs_[core_id];

        apex::core::cross_core_post(
            engine_, core_id,
            [mgr, session_ids = std::move(session_ids),
             shared_data]() {
                for (auto sid : session_ids) {
                    auto session = mgr->find(sid);
                    if (!session || !session->is_open()) continue;
                    session->enqueue_write_raw(*shared_data);
                }
            });
    }
}

} // namespace apex::gateway
```
