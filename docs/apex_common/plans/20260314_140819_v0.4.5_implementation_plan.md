```cpp
// apex_core/include/apex/core/core_allocator.hpp
#pragma once

#include <concepts>
#include <cstddef>

namespace apex::core {

/// 모든 할당기가 만족하는 공통 concept.
template <typename T>
concept CoreAllocator = requires(T a, void* ptr) {
    { a.allocate(std::size_t{}, std::size_t{}) } -> std::same_as<void*>;
    { a.owns(ptr) }       -> std::same_as<bool>;
    { a.used_bytes() }    -> std::convertible_to<std::size_t>;
    { a.capacity() }      -> std::convertible_to<std::size_t>;
};

/// 개별 해제 가능 (SlabAllocator).
template <typename T>
concept Freeable = CoreAllocator<T> && requires(T a, void* ptr) {
    { a.deallocate(ptr) } -> std::same_as<void>;
};

/// 벌크 리셋 가능 (BumpAllocator, ArenaAllocator).
template <typename T>
concept Resettable = CoreAllocator<T> && requires(T a) {
    { a.reset() } -> std::same_as<void>;
};

} // namespace apex::core
```

```cpp
// apex_core/include/apex/core/bump_allocator.hpp
#pragma once

#include <cstddef>
#include <cstdint>

namespace apex::core {

/// 요청/코루틴 수명 임시 데이터용 bump 할당기.
/// 개별 해제 없음 — reset()으로 전체 초기화.
/// thread-safety 불필요 (코어당 1개, io_context 보장).
class BumpAllocator {
public:
    /// @param capacity 청크 바이트 수 (기동 시 malloc 1회)
    explicit BumpAllocator(std::size_t capacity);
    ~BumpAllocator();

    BumpAllocator(const BumpAllocator&) = delete;
    BumpAllocator& operator=(const BumpAllocator&) = delete;
    BumpAllocator(BumpAllocator&& other) noexcept;
    BumpAllocator& operator=(BumpAllocator&& other) noexcept;

    /// alignment 경계로 올린 후 size만큼 전진. 초과 시 nullptr.
    [[nodiscard]] void* allocate(std::size_t size,
                                 std::size_t align = alignof(std::max_align_t));

    /// cursor를 시작 위치로 복원 (O(1)).
    void reset() noexcept;

    /// ptr이 이 할당기의 청크 내부인지 확인.
    [[nodiscard]] bool owns(void* ptr) const noexcept;

    /// 현재 사용량 (cursor - base).
    [[nodiscard]] std::size_t used_bytes() const noexcept;

    /// 전체 용량.
    [[nodiscard]] std::size_t capacity() const noexcept;

private:
    char* base_{nullptr};     // malloc'd chunk 시작
    char* cursor_{nullptr};   // 다음 할당 위치
    char* end_{nullptr};      // chunk 끝 (base + capacity)
    std::size_t capacity_{0};
};

} // namespace apex::core
```

```cpp
// apex_core/src/bump_allocator.cpp
#include <apex/core/bump_allocator.hpp>
#include <cstdlib>
#include <cstring>
#include <utility>

namespace apex::core {

BumpAllocator::BumpAllocator(std::size_t capacity)
    : capacity_(capacity) {
    if (capacity_ > 0) {
        base_ = static_cast<char*>(std::malloc(capacity_));
        cursor_ = base_;
        end_ = base_ + capacity_;
    }
}

BumpAllocator::~BumpAllocator() {
    std::free(base_);
}

BumpAllocator::BumpAllocator(BumpAllocator&& other) noexcept
    : base_(std::exchange(other.base_, nullptr))
    , cursor_(std::exchange(other.cursor_, nullptr))
    , end_(std::exchange(other.end_, nullptr))
    , capacity_(std::exchange(other.capacity_, 0)) {}

BumpAllocator& BumpAllocator::operator=(BumpAllocator&& other) noexcept {
    if (this != &other) {
        std::free(base_);
        base_ = std::exchange(other.base_, nullptr);
        cursor_ = std::exchange(other.cursor_, nullptr);
        end_ = std::exchange(other.end_, nullptr);
        capacity_ = std::exchange(other.capacity_, 0);
    }
    return *this;
}

void* BumpAllocator::allocate(std::size_t size, std::size_t align) {
    if (size == 0) return nullptr;
    // cursor를 alignment 경계로 올림
    auto addr = reinterpret_cast<uintptr_t>(cursor_);
    auto aligned = (addr + align - 1) & ~(align - 1);
    auto* result = reinterpret_cast<char*>(aligned);
    auto* new_cursor = result + size;
    if (new_cursor > end_) return nullptr;
    cursor_ = new_cursor;
    return result;
}

void BumpAllocator::reset() noexcept {
    cursor_ = base_;
}

bool BumpAllocator::owns(void* ptr) const noexcept {
    auto* p = static_cast<char*>(ptr);
    return p >= base_ && p < end_;
}

std::size_t BumpAllocator::used_bytes() const noexcept {
    return static_cast<std::size_t>(cursor_ - base_);
}

std::size_t BumpAllocator::capacity() const noexcept {
    return capacity_;
}

} // namespace apex::core
```

```cpp
// apex_core/include/apex/core/arena_allocator.hpp
#pragma once

#include <cstddef>
#include <cstdint>
#include <vector>

namespace apex::core {

/// 트랜잭션 수명 데이터용 아레나 할당기.
/// 블록 체이닝 + bump 방식. reset()으로 벌크 해제.
/// thread-safety 불필요 (코어당 1개).
class ArenaAllocator {
public:
    static constexpr std::size_t kDefaultBlockSize = 4096;
    static constexpr std::size_t kDefaultMaxBytes = 1024 * 1024;

    /// @param block_size 블록 크기 (default 4KB)
    /// @param max_bytes 전체 블록 합계 상한 (default 1MB)
    explicit ArenaAllocator(std::size_t block_size = kDefaultBlockSize,
                            std::size_t max_bytes = kDefaultMaxBytes);
    ~ArenaAllocator();

    ArenaAllocator(const ArenaAllocator&) = delete;
    ArenaAllocator& operator=(const ArenaAllocator&) = delete;
    ArenaAllocator(ArenaAllocator&&) noexcept;
    ArenaAllocator& operator=(ArenaAllocator&&) noexcept;

    /// 블록 내 bump 할당. 블록 초과 시 새 블록 추가.
    /// max_bytes 초과 시 nullptr.
    [[nodiscard]] void* allocate(std::size_t size,
                                 std::size_t align = alignof(std::max_align_t));

    /// 전체 블록 체인 해제 (첫 블록만 유지 재사용).
    void reset() noexcept;

    /// ptr이 어느 블록에든 속하는지 확인.
    [[nodiscard]] bool owns(void* ptr) const noexcept;

    /// 전체 할당 바이트 (패딩 포함).
    [[nodiscard]] std::size_t used_bytes() const noexcept;

    /// 현재 확보된 총 용량.
    [[nodiscard]] std::size_t capacity() const noexcept;

private:
    struct Block {
        char* base;
        char* cursor;
        char* end;
        std::size_t size;
    };

    Block allocate_block(std::size_t size);
    void free_block(Block& block) noexcept;

    std::vector<Block> blocks_;
    std::size_t block_size_;
    std::size_t max_bytes_;
    std::size_t total_allocated_{0};  // 전체 블록 합계
};

} // namespace apex::core
```

```cpp
// apex_core/src/arena_allocator.cpp
#include <apex/core/arena_allocator.hpp>
#include <algorithm>
#include <cstdlib>
#include <utility>

namespace apex::core {

ArenaAllocator::ArenaAllocator(std::size_t block_size, std::size_t max_bytes)
    : block_size_(block_size), max_bytes_(max_bytes) {
    blocks_.push_back(allocate_block(block_size_));
    total_allocated_ = block_size_;
}

ArenaAllocator::~ArenaAllocator() {
    for (auto& block : blocks_) {
        free_block(block);
    }
}

ArenaAllocator::ArenaAllocator(ArenaAllocator&& other) noexcept
    : blocks_(std::move(other.blocks_))
    , block_size_(other.block_size_)
    , max_bytes_(other.max_bytes_)
    , total_allocated_(other.total_allocated_) {
    other.total_allocated_ = 0;
}

ArenaAllocator& ArenaAllocator::operator=(ArenaAllocator&& other) noexcept {
    if (this != &other) {
        for (auto& block : blocks_) free_block(block);
        blocks_ = std::move(other.blocks_);
        block_size_ = other.block_size_;
        max_bytes_ = other.max_bytes_;
        total_allocated_ = other.total_allocated_;
        other.total_allocated_ = 0;
    }
    return *this;
}

void* ArenaAllocator::allocate(std::size_t size, std::size_t align) {
    if (size == 0) return nullptr;

    // 현재 블록에서 시도
    auto& current = blocks_.back();
    auto addr = reinterpret_cast<uintptr_t>(current.cursor);
    auto aligned = (addr + align - 1) & ~(align - 1);
    auto* result = reinterpret_cast<char*>(aligned);
    auto* new_cursor = result + size;

    if (new_cursor <= current.end) {
        current.cursor = new_cursor;
        return result;
    }

    // 새 블록 필요
    auto new_block_size = std::max(block_size_, size + align);
    if (total_allocated_ + new_block_size > max_bytes_) {
        return nullptr;  // 상한 초과
    }

    blocks_.push_back(allocate_block(new_block_size));
    total_allocated_ += new_block_size;

    auto& fresh = blocks_.back();
    addr = reinterpret_cast<uintptr_t>(fresh.cursor);
    aligned = (addr + align - 1) & ~(align - 1);
    result = reinterpret_cast<char*>(aligned);
    fresh.cursor = result + size;
    return result;
}

void ArenaAllocator::reset() noexcept {
    // 첫 블록만 유지, 나머지 해제
    for (std::size_t i = 1; i < blocks_.size(); ++i) {
        free_block(blocks_[i]);
    }
    blocks_.resize(1);
    blocks_[0].cursor = blocks_[0].base;
    total_allocated_ = blocks_[0].size;
}

bool ArenaAllocator::owns(void* ptr) const noexcept {
    auto* p = static_cast<char*>(ptr);
    for (const auto& block : blocks_) {
        if (p >= block.base && p < block.end) return true;
    }
    return false;
}

std::size_t ArenaAllocator::used_bytes() const noexcept {
    std::size_t total = 0;
    for (const auto& block : blocks_) {
        total += static_cast<std::size_t>(block.cursor - block.base);
    }
    return total;
}

std::size_t ArenaAllocator::capacity() const noexcept {
    return total_allocated_;
}

ArenaAllocator::Block ArenaAllocator::allocate_block(std::size_t size) {
    auto* base = static_cast<char*>(std::malloc(size));
    return {base, base, base + size, size};
}

void ArenaAllocator::free_block(Block& block) noexcept {
    std::free(block.base);
    block.base = nullptr;
}

} // namespace apex::core
```

```cpp
namespace apex::shared::adapters::redis {

RedisMultiplexer::RedisMultiplexer(boost::asio::io_context& io_ctx,
                                   const RedisConfig& config)
    : io_ctx_(io_ctx), config_(config) {}

RedisMultiplexer::~RedisMultiplexer() {
    // pending 명령에 에러 설정
    for (auto* cmd : pending_) {
        cmd->result = apex::core::ErrorCode::AdapterError;
        cmd->resolver.cancel();
    }
}

awaitable<Result<RedisReply>> RedisMultiplexer::command(std::string_view cmd) {
    if (!conn_ || !conn_->is_connected()) {
        co_return apex::core::ErrorCode::AdapterError;
    }

    auto seq = next_sequence_++;
    PendingCommand pending{
        .resolver = boost::asio::steady_timer(io_ctx_, std::chrono::steady_clock::time_point::max()),
        .timeout = boost::asio::steady_timer(io_ctx_, config_.command_timeout),
        .result = apex::core::ErrorCode::Timeout,
        .sequence = seq
    };

    pending_.push_back(&pending);

    // hiredis 비동기 명령 전송
    // RedisConnection에 async_context() public accessor 추가 필요
    std::string cmd_str(cmd);
    int ret = redisAsyncCommand(conn_->async_context(), &RedisMultiplexer::static_on_reply,
                                this, cmd_str.c_str());
    if (ret != REDIS_OK) {
        pending_.pop_back();
        co_return apex::core::ErrorCode::AdapterError;
    }

    // 타임아웃 설정 — 만료 시 에러로 resume
    pending.timeout.async_wait([&pending, seq](boost::system::error_code ec) {
        if (!ec) {
            pending.result = apex::core::ErrorCode::Timeout;
            pending.resolver.cancel();
        }
    });

    // 응답 대기 (hiredis callback이 resolver.cancel()로 resume)
    boost::system::error_code ec;
    co_await pending.resolver.async_wait(boost::asio::redirect_error(
        boost::asio::use_awaitable, ec));

    pending.timeout.cancel();
    co_return std::move(pending.result);
}

// hiredis callback → FIFO 매칭
void RedisMultiplexer::static_on_reply(redisAsyncContext* /*ac*/, void* reply, void* privdata) {
    auto* self = static_cast<RedisMultiplexer*>(privdata);
    auto* r = static_cast<redisReply*>(reply);

    if (self->pending_.empty()) return;  // 타임아웃으로 이미 제거된 경우

    auto* front = self->pending_.front();
    self->pending_.pop_front();

    if (r && r->type != REDIS_REPLY_ERROR) {
        front->result = RedisReply{r};  // RedisReply가 redisReply를 래핑 (복사)
    } else {
        front->result = apex::core::ErrorCode::AdapterError;
    }
    front->resolver.cancel();  // 코루틴 resume

    // 타임아웃-응답 경합 주의: 타임아웃된 명령은 pending_에서 이미 제거되므로,
    // hiredis 응답이 FIFO로 도착하면 다음 pending과 불일치할 수 있음.
    // 해결: 타임아웃 시 pending_에서 제거하지 않고 result만 설정 + flag.
    // 이후 이 callback에서 flag 확인하여 skip.
    // TODO: v0.4.5 구현 시 타임아웃-FIFO 정합성 상세 구현 필요
}

void RedisMultiplexer::on_disconnect() {
    reconnecting_ = true;
    // 대기 중인 모든 명령에 에러 반환
    for (auto* cmd : pending_) {
        cmd->result = apex::core::ErrorCode::AdapterError;
        cmd->resolver.cancel();
    }
    pending_.clear();
    // 재연결 루프 시작
    boost::asio::co_spawn(io_ctx_, reconnect_loop(), boost::asio::detached);
}

awaitable<void> RedisMultiplexer::reconnect_loop() {
    auto backoff = std::chrono::milliseconds{100};
    const auto max_backoff = config_.reconnect_max_backoff;

    while (reconnecting_) {
        ++reconnect_attempts_;
        spdlog::info("Redis reconnect attempt {} (backoff {}ms)",
                     reconnect_attempts_, backoff.count());

        // 재연결 시도 — create()는 unique_ptr 반환 (null이면 실패)
        conn_ = RedisConnection::create(io_ctx_, config_);
        if (conn_ && conn_->is_connected()) {
            reconnecting_ = false;
            reconnect_attempts_ = 0;
            spdlog::info("Redis reconnected successfully");
            co_return;
        }

        // 지수 백오프 대기
        boost::asio::steady_timer timer(io_ctx_, backoff);
        co_await timer.async_wait(boost::asio::use_awaitable);
        backoff = std::min(backoff * 2, max_backoff);
    }
}

bool RedisMultiplexer::connected() const noexcept {
    return conn_ && conn_->is_connected() && !reconnecting_;
}

std::size_t RedisMultiplexer::pending_count() const noexcept {
    return pending_.size();
}

uint32_t RedisMultiplexer::reconnect_attempts() const noexcept {
    return reconnect_attempts_;
}
```

```cpp
    auto& target = *cores_[target_core];
    target.metrics.post_total.fetch_add(1, std::memory_order_relaxed);

    auto result = target.inbox->enqueue(msg);
    if (!result) {
        target.metrics.post_failures.fetch_add(1, std::memory_order_relaxed);
        // Rate-limited 경고 (1초당 최대 1회)

            spdlog::warn("cross_core_post to core {} failed: queue full", target_core);
            last_log = now;
        }
        return ErrorCode::CrossCoreQueueFull;
    }

}
```
