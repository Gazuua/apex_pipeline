# Plan 3: Auth Service — JWT 발급 + Redis 세션 + Refresh Token

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Kafka 경유 로그인 → JWT 발급 → Redis 세션 → Refresh Token 갱신 플로우 구현

**Architecture:** ServiceBase 상속, Kafka consumer/producer로 Gateway와 통신. PostgreSQL(auth_svc 스키마)에 유저 DB, Redis에 세션/블랙리스트.

**Tech Stack:** C++23, jwt-cpp, bcrypt, librdkafka, hiredis, libpq, FlatBuffers

---

```cpp
void AuthService::start() {
    spdlog::info("[AuthService] Starting...");

    // Kafka consumer 콜백 등록
    kafka_.set_message_callback(
        [this](std::string_view topic, int32_t partition,
               std::span<const uint8_t> key,
               std::span<const uint8_t> payload,
               int64_t offset) -> apex::core::Result<void> {
```

```sql
CREATE SCHEMA IF NOT EXISTS auth_svc;

-- 전용 롤 생성 (cross-schema 접근 차단)
DO $$
BEGIN
    IF NOT EXISTS (SELECT FROM pg_roles WHERE rolname = 'auth_role') THEN
        CREATE ROLE auth_role WITH LOGIN PASSWORD 'auth_secret_change_me';
    END IF;
END
$$;

-- 권한 부여: auth_svc 스키마에만 접근 허용
GRANT USAGE ON SCHEMA auth_svc TO auth_role;
GRANT CREATE ON SCHEMA auth_svc TO auth_role;

-- 기본 권한: auth_svc 스키마에 생성되는 모든 테이블에 대해 자동 부여
ALTER DEFAULT PRIVILEGES IN SCHEMA auth_svc
    GRANT SELECT, INSERT, UPDATE, DELETE ON TABLES TO auth_role;

-- public 스키마 접근 차단
REVOKE ALL ON SCHEMA public FROM auth_role;

COMMENT ON SCHEMA auth_svc IS 'Auth Service 전용 스키마 — users, refresh_tokens';
```

```sql
-- users 테이블 — 사용자 인증 정보
SET search_path TO auth_svc;

CREATE TABLE IF NOT EXISTS users (
    id              BIGSERIAL       PRIMARY KEY,
    email           VARCHAR(255)    NOT NULL UNIQUE,
    password_hash   VARCHAR(72)     NOT NULL,  -- bcrypt 해시 (60자) + 여유
    created_at      TIMESTAMPTZ     NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ     NOT NULL DEFAULT NOW(),
    locked_until    TIMESTAMPTZ,               -- NULL이면 잠기지 않음
    failed_attempts SMALLINT        NOT NULL DEFAULT 0,

    -- 인덱스: email 검색 최적화 (UNIQUE 제약에 의해 자동 생성됨)
    CONSTRAINT users_email_check CHECK (email ~* '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$')
);

-- updated_at 자동 갱신 트리거
CREATE OR REPLACE FUNCTION auth_svc.update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_users_updated_at
    BEFORE UPDATE ON users
    FOR EACH ROW
    EXECUTE FUNCTION auth_svc.update_updated_at();

COMMENT ON TABLE users IS '사용자 인증 정보. password_hash는 bcrypt 해시.';
COMMENT ON COLUMN users.locked_until IS 'NULL이면 활성 상태. 날짜 설정 시 해당 시점까지 잠금.';
COMMENT ON COLUMN users.failed_attempts IS '연속 실패 횟수. 성공 시 0으로 초기화.';
```

### Step 2.4: refresh_tokens 테이블

```sql
-- refresh_tokens 테이블 — Refresh Token 관리
SET search_path TO auth_svc;

CREATE TABLE IF NOT EXISTS refresh_tokens (
    id              BIGSERIAL       PRIMARY KEY,
    token_hash      VARCHAR(64)     NOT NULL UNIQUE,  -- SHA-256 해시 (hex, 64자)
    user_id         BIGINT          NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    expires_at      TIMESTAMPTZ     NOT NULL,
    created_at      TIMESTAMPTZ     NOT NULL DEFAULT NOW(),
    revoked_at      TIMESTAMPTZ,                      -- NULL이면 활성, 설정 시 폐기됨
    replaced_by     BIGINT          REFERENCES refresh_tokens(id),  -- 갱신 시 새 토큰 ID

    -- 인덱스: user_id로 빠른 조회 (활성 토큰 목록)
    CONSTRAINT refresh_tokens_expiry_check CHECK (expires_at > created_at)
);

-- user_id 기반 조회 인덱스
CREATE INDEX IF NOT EXISTS idx_refresh_tokens_user_id
    ON refresh_tokens(user_id)
    WHERE revoked_at IS NULL;

-- 만료된 토큰 정리용 인덱스
CREATE INDEX IF NOT EXISTS idx_refresh_tokens_expires_at
    ON refresh_tokens(expires_at)
    WHERE revoked_at IS NULL;

COMMENT ON TABLE refresh_tokens IS 'Refresh Token 관리. token_hash는 SHA-256(token)의 hex.';
COMMENT ON COLUMN refresh_tokens.replaced_by IS '토큰 갱신(rotation) 시 새 토큰의 ID. Refresh Token Reuse Detection에 활용.';
```

```cpp
#include <bcrypt.h>  // bcrypt_hashpw, bcrypt_gensalt, bcrypt_checkpw

    : work_factor_(work_factor)

std::string PasswordHasher::hash(std::string_view password) const {
    char salt[BCRYPT_HASHSIZE];
    char hash[BCRYPT_HASHSIZE];

    if (bcrypt_gensalt(static_cast<int>(work_factor_), salt) != 0) {
        spdlog::error("[PasswordHasher] bcrypt_gensalt failed");
        return {};
    }

    if (bcrypt_hashpw(std::string(password).c_str(), salt, hash) != 0) {
        spdlog::error("[PasswordHasher] bcrypt_hashpw failed");
        return {};
    }

    return std::string(hash);
}

bool PasswordHasher::verify(std::string_view password,
                            std::string_view stored_hash) const {
    // bcrypt_checkpw: 0이면 일치
    int ret = bcrypt_checkpw(std::string(password).c_str(),
                             std::string(stored_hash).c_str());
    return ret == 0;
}
```

```cpp
#include <apex/auth_svc/jwt_manager.hpp>

#include <jwt-cpp/jwt.h>
#include <spdlog/spdlog.h>

#include <fstream>
#include <sstream>

namespace apex::auth_svc {

namespace {

std::string read_file(std::string_view path) {
    std::ifstream file(std::string(path));
    if (!file.is_open()) {
        spdlog::error("[JwtManager] Failed to open key file: {}", path);
        return {};
    }
    std::ostringstream ss;
    ss << file.rdbuf();
    return ss.str();
}

} // anonymous namespace

JwtManager::JwtManager(std::string_view private_key_path,
                       std::string_view public_key_path,
                       std::string_view issuer,
                       std::chrono::seconds access_token_ttl)
    : private_key_(read_file(private_key_path))
    , public_key_(read_file(public_key_path))
    , issuer_(issuer)
    , access_token_ttl_(access_token_ttl)
{
    if (private_key_.empty()) {
        spdlog::warn("[JwtManager] Private key not loaded — token creation disabled");
    }
    if (public_key_.empty()) {
        spdlog::warn("[JwtManager] Public key not loaded — token verification disabled");
    }
}

std::string JwtManager::create_access_token(uint64_t user_id,
                                             std::string_view email) const {
    if (private_key_.empty()) {
        spdlog::error("[JwtManager] Cannot create token — private key not loaded");
        return {};
    }

    auto now = std::chrono::system_clock::now();
    auto exp = now + access_token_ttl_;

    auto token = jwt::create()
        .set_issuer(issuer_)
        .set_type("JWT")
        .set_issued_at(now)
        .set_expires_at(exp)
        .set_payload_claim("uid", jwt::claim(std::to_string(user_id)))
        .set_payload_claim("email", jwt::claim(std::string(email)))
        .sign(jwt::algorithm::rs256(public_key_, private_key_));

    return token;
}

apex::core::Result<JwtManager::Claims> JwtManager::verify_access_token(
    std::string_view token) const
{
    if (public_key_.empty()) {
        return apex::core::error(apex::core::ErrorCode::AdapterError);
    }

    try {
        auto verifier = jwt::verify()
            .allow_algorithm(jwt::algorithm::rs256(public_key_, private_key_))
            .with_issuer(issuer_);

        auto decoded = jwt::decode(std::string(token));
        verifier.verify(decoded);

        Claims claims;
        claims.user_id = std::stoull(decoded.get_payload_claim("uid").as_string());
        claims.email = decoded.get_payload_claim("email").as_string();
        claims.issued_at = decoded.get_issued_at();
        claims.expires_at = decoded.get_expires_at();

        return claims;

    } catch (const jwt::error::token_verification_exception& e) {
        spdlog::debug("[JwtManager] Token verification failed: {}", e.what());
        return apex::core::error(apex::core::ErrorCode::AdapterError);
    } catch (const std::exception& e) {
        spdlog::error("[JwtManager] Unexpected error: {}", e.what());
        return apex::core::error(apex::core::ErrorCode::AdapterError);
    }
}

std::chrono::seconds JwtManager::remaining_ttl(std::string_view token) const {
    try {
        auto decoded = jwt::decode(std::string(token));
        auto exp = decoded.get_expires_at();
        auto now = std::chrono::system_clock::now();

        if (exp <= now) {
            return std::chrono::seconds{0};
        }

        return std::chrono::duration_cast<std::chrono::seconds>(exp - now);
    } catch (...) {
        return std::chrono::seconds{0};
    }
}
```

```cpp
#include <cstdint>
#include <cstring>

namespace apex::auth_svc {

/// Kafka Envelope 구조 (설계서 §2.2 기준)
///
/// Routing Header (8B, 불변):
///   header_version(u16) | flags(u16) | msg_id(u32)
///
/// Metadata Prefix (32B, 고정):
///   meta_version(u32) | core_id(u16) | corr_id(u64) |
///   source_id(u16) | session_id(u64) | timestamp(u64)
///
/// FlatBuffers Payload: offset 40B~

constexpr size_t ROUTING_HEADER_SIZE = 8;
constexpr size_t METADATA_SIZE = 32;
constexpr size_t ENVELOPE_HEADER_SIZE = ROUTING_HEADER_SIZE + METADATA_SIZE;  // 40B

struct RoutingHeader {
    uint16_t header_version{1};
    uint16_t flags{0};
    uint32_t msg_id{0};
};

struct EnvelopeMetadata {
    uint32_t meta_version{1};
    uint16_t core_id{0};
    uint64_t corr_id{0};
    uint16_t source_id{0};
    uint64_t session_id{0};
    uint64_t timestamp{0};
};

struct ParsedEnvelope {
    RoutingHeader routing;
    EnvelopeMetadata metadata;
    std::span<const uint8_t> payload;  // FlatBuffers payload (offset 40~)
};

/// Kafka Envelope 파싱
[[nodiscard]] inline apex::core::Result<ParsedEnvelope>
parse_envelope(std::span<const uint8_t> data) {
    if (data.size() < ENVELOPE_HEADER_SIZE) {
        return apex::core::error(apex::core::ErrorCode::FlatBuffersVerifyFailed);
    }

    ParsedEnvelope env;
    size_t offset = 0;

    // Routing Header (8B) — big-endian
    auto read_u16 = [&](size_t off) -> uint16_t {
        uint16_t val;
        std::memcpy(&val, data.data() + off, sizeof(uint16_t));
        return (val >> 8) | (val << 8);  // ntohs
    };
    auto read_u32 = [&](size_t off) -> uint32_t {
        uint32_t val;
        std::memcpy(&val, data.data() + off, sizeof(uint32_t));
        return __builtin_bswap32(val);  // ntohl
    };
    auto read_u64 = [&](size_t off) -> uint64_t {
        uint64_t val;
        std::memcpy(&val, data.data() + off, sizeof(uint64_t));
        return __builtin_bswap64(val);  // ntohll
    };

    env.routing.header_version = read_u16(0);
    env.routing.flags = read_u16(2);
    env.routing.msg_id = read_u32(4);

    // Metadata Prefix (32B)
    offset = ROUTING_HEADER_SIZE;
    env.metadata.meta_version = read_u32(offset);
    env.metadata.core_id = read_u16(offset + 4);
    env.metadata.corr_id = read_u64(offset + 6);
    env.metadata.source_id = read_u16(offset + 14);
    env.metadata.session_id = read_u64(offset + 16);
    env.metadata.timestamp = read_u64(offset + 24);

    // Payload
    env.payload = data.subspan(ENVELOPE_HEADER_SIZE);

    return env;
}

/// Kafka Envelope 조립 (응답용)
[[nodiscard]] inline std::vector<uint8_t>
build_envelope(const RoutingHeader& routing,
               const EnvelopeMetadata& metadata,
               std::span<const uint8_t> payload)
{
    std::vector<uint8_t> buf(ENVELOPE_HEADER_SIZE + payload.size());

    auto write_u16 = [&](size_t off, uint16_t val) {
        uint16_t net = (val >> 8) | (val << 8);
        std::memcpy(buf.data() + off, &net, sizeof(uint16_t));
    };
    auto write_u32 = [&](size_t off, uint32_t val) {
        uint32_t net = __builtin_bswap32(val);
        std::memcpy(buf.data() + off, &net, sizeof(uint32_t));
    };
    auto write_u64 = [&](size_t off, uint64_t val) {
        uint64_t net = __builtin_bswap64(val);
        std::memcpy(buf.data() + off, &net, sizeof(uint64_t));
    };

    // Routing Header
    write_u16(0, routing.header_version);
    write_u16(2, routing.flags);
    write_u32(4, routing.msg_id);

    // Metadata
    size_t off = ROUTING_HEADER_SIZE;
    write_u32(off, metadata.meta_version);
    write_u16(off + 4, metadata.core_id);
    write_u64(off + 6, metadata.corr_id);
    write_u16(off + 14, metadata.source_id);
    write_u64(off + 16, metadata.session_id);
    write_u64(off + 24, metadata.timestamp);

    // Payload
    std::memcpy(buf.data() + ENVELOPE_HEADER_SIZE, payload.data(), payload.size());

    return buf;
}

/// Routing Header flags 비트 조작 헬퍼
namespace envelope_flags {
    constexpr uint16_t DIRECTION_RESPONSE = 0x0001;  // bit 0
    constexpr uint16_t ERROR_BIT          = 0x0002;  // bit 1

    [[nodiscard]] constexpr bool is_response(uint16_t flags) noexcept {
        return (flags & DIRECTION_RESPONSE) != 0;
    }
    [[nodiscard]] constexpr bool is_error(uint16_t flags) noexcept {
        return (flags & ERROR_BIT) != 0;
    }
    [[nodiscard]] constexpr uint16_t set_response(uint16_t flags) noexcept {
        return flags | DIRECTION_RESPONSE;
    }
    [[nodiscard]] constexpr uint16_t set_error(uint16_t flags) noexcept {
        return flags | ERROR_BIT;
    }
} // namespace envelope_flags

/// Auth Service source_id (설계서 §2.2: 1=Auth)
constexpr uint16_t AUTH_SOURCE_ID = 1;

} // namespace apex::auth_svc
```

> **MSVC 호환성 주의**: `__builtin_bswap32`/`__builtin_bswap64`는 GCC/Clang 내장 함수. MSVC에서는 `_byteswap_ulong`/`_byteswap_uint64` 사용. 빌드 시 `#ifdef _MSC_VER` 분기 추가 필요.

```cpp
    #define bswap64(x) _byteswap_uint64(x)

    #define bswap32(x) __builtin_bswap32(x)
    #define bswap64(x) __builtin_bswap64(x)
```

```cpp
namespace apex::auth_svc {

/// SHA-256 해시 → hex 문자열 (64자)
[[nodiscard]] inline std::string sha256_hex(std::string_view input) {
    std::array<uint8_t, SHA256_DIGEST_LENGTH> hash{};
    SHA256(reinterpret_cast<const uint8_t*>(input.data()),
           input.size(),
           hash.data());

    std::string result;
    result.reserve(SHA256_DIGEST_LENGTH * 2);
    for (auto byte : hash) {
        result += std::format("{:02x}", byte);
    }
    return result;
}

/// 암호학적으로 안전한 랜덤 토큰 생성 (hex, 64자)
[[nodiscard]] inline std::string generate_secure_token(size_t bytes = 32) {
    std::vector<uint8_t> buf(bytes);
    RAND_bytes(buf.data(), static_cast<int>(bytes));

    std::string result;
    result.reserve(bytes * 2);
    for (auto byte : buf) {
        result += std::format("{:02x}", byte);
    }
    return result;
}

} // namespace apex::auth_svc
```

### Step 6.2: Refresh Token Rotation 정책 문서화

> **Refresh Token Rotation 규칙:**
>
> 1. 갱신 시 항상 새 Refresh Token 발급 (1회용)
> 2. 기존 Refresh Token은 즉시 폐기 (`revoked_at` 설정)
> 3. 폐기된 토큰이 재사용되면 → **해당 user의 모든 활성 Refresh Token 폐기** (Reuse Detection)
> 4. 클라이언트는 새 Refresh Token을 반드시 저장해야 함
> 5. 모든 Refresh Token이 폐기되면 재로그인 필요

## 주요 설계 결정 메모

1. **ServiceBase 미상속**: Auth Service는 Kafka consumer 기반이지 세션 기반이 아니므로, `ServiceBase<T>` CRTP를 상속하지 않고 독립 클래스로 구현. Kafka 콜백 → 코루틴 spawn → PG/Redis 비동기 호출 패턴.

2. **Kafka 콜백 → 코루틴 브릿지**: `MessageCallback`은 동기(`Result<void>` 반환)인데 PG/Redis는 코루틴. `boost::asio::co_spawn(io_ctx, handler_coroutine, detached)` 패턴으로 연결. 콜백에서는 즉시 `ok()` 반환, 실제 처리는 spawn된 코루틴에서 수행.

3. **Refresh Token Rotation**: 갱신 시마다 새 토큰 발급 + 기존 폐기. 폐기된 토큰 재사용 감지(Reuse Detection) → 모든 토큰 일괄 폐기.

4. **MSVC 바이트스왑**: `__builtin_bswap32/64` 대신 `_byteswap_ulong`/`_byteswap_uint64` 사용. `#ifdef _MSC_VER` 분기 필수.

5. **bcrypt work factor**: 테스트 시 4, 프로덕션 12. config로 분리.
