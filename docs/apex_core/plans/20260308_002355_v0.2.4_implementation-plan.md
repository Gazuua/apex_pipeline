### Step 3: core_engine.cpp에 start()/join() 구현

`src/core_engine.cpp` 수정 — 기존 `run()`을 `start()` + `join()`으로 분리:

```cpp
void CoreEngine::start() {
    running_.store(true, std::memory_order_release);

    for (uint32_t i = 0; i < config_.num_cores; ++i) {
        threads_.emplace_back([this, i]() { run_core(i); });
    }
}

void CoreEngine::join() {
    for (auto& t : threads_) {
        if (t.joinable()) t.join();
    }
    threads_.clear();
    running_.store(false, std::memory_order_release);
}

void CoreEngine::run() {
    start();
    join();
}
```

주의: 기존 `run()` 내부의 로직을 확인하고, `running_` 플래그 세팅 타이밍이 동일하도록 보장.
기존 `run()`에서 `running_` 세팅하는 위치를 확인한 뒤 `start()`에서 동일하게 처리할 것.

```cpp
struct CoreMessage {
    enum class Type : uint8_t {
        Shutdown = 0,
        DrainComplete,
        Custom,
        CrossCoreRequest,    // cross_core_call 요청
        CrossCorePost,       // cross_core_post fire-and-forget
    };
    // ... 기존 동일 ...
};

class CoreEngine {
public:
    using DrainCallback = std::function<void(uint32_t core_id)>;

    // ... 기존 ...

    /// Set callback invoked on each drain cycle per core.
    void set_drain_callback(DrainCallback callback);

    /// Drain remaining messages from all inboxes, cleaning up heap pointers.
    /// Call after stop() and before join().
    void drain_remaining();

private:
    // ... 기존 ...
    DrainCallback drain_callback_;
};
```

`src/core_engine.cpp` 수정:

- `set_drain_callback()` 구현 (단순 멤버 할당)
- `start_drain_timer()` 내부에서 drain 루프 후 `drain_callback_(core_id)` 호출
- `drain_remaining()`: 모든 코어의 inbox를 drain하면서 CrossCoreRequest/CrossCorePost 타입의 data를 `delete reinterpret_cast<void*>(msg.data)`로 정리

```cpp
void CoreEngine::set_drain_callback(DrainCallback callback) {
    drain_callback_ = std::move(callback);
}

void CoreEngine::drain_remaining() {
    for (auto& ctx : cores_) {
        while (auto msg = ctx->inbox->dequeue()) {
            if (msg->type == CoreMessage::Type::CrossCoreRequest ||
                msg->type == CoreMessage::Type::CrossCorePost) {
                // CrossCoreTask* 또는 유사 힙 객체 정리
                delete reinterpret_cast<char*>(msg->data);  // 타입 소거된 정리
            }
        }
    }
}
```

주의: `drain_remaining()`의 delete 대상 타입은 Task 8(cross_core_call)에서 `CrossCoreTask`로 구체화됨.
이 시점에서는 포인터 정리 프로토콜만 확립하고, 실제 타입은 Task 8에서 맞춘다.

`start_drain_timer()` 수정:
```cpp
void CoreEngine::start_drain_timer(uint32_t core_id) {
    auto& ctx = *cores_[core_id];
    ctx.drain_timer->expires_after(config_.drain_interval);
    ctx.drain_timer->async_wait([this, core_id](boost::system::error_code ec) {
        if (ec || !running_.load(std::memory_order_relaxed)) return;

        auto& ctx = *cores_[core_id];
        while (auto msg = ctx.inbox->dequeue()) {
            if (message_handler_) {
                message_handler_(core_id, *msg);
            }
        }

        // drain 콜백 호출 (tick 등)
        if (drain_callback_) {
            drain_callback_(core_id);
        }

        start_drain_timer(core_id);
    });
}
```

참조: 설계 문서의 "Server API" 섹션.

```cpp
#pragma once

#include <apex/core/core_engine.hpp>
#include <apex/core/message_dispatcher.hpp>
#include <apex/core/session_manager.hpp>
#include <apex/core/service_base.hpp>
#include <apex/core/tcp_acceptor.hpp>

#include <boost/asio/awaitable.hpp>
#include <boost/asio/io_context.hpp>
#include <boost/asio/signal_set.hpp>
#include <boost/asio/steady_timer.hpp>

#include <atomic>
#include <cstdint>
#include <functional>
#include <memory>
#include <string>
#include <vector>

namespace apex::core {

struct ServerConfig {
    // 네트워크
    uint16_t port = 9000;
    std::string bind_address = "0.0.0.0";

    // 멀티코어
    uint32_t num_cores = 1;
    size_t mpsc_queue_capacity = 65536;
    std::chrono::microseconds drain_interval{100};

    // 세션
    uint32_t heartbeat_timeout_ticks = 300;
    uint32_t tick_interval_ms = 100;
    size_t recv_buf_capacity = 8192;
    size_t timer_wheel_slots = 1024;

    // 시그널
    bool handle_signals = true;
};

struct PerCoreState {
    uint32_t core_id;
    SessionManager session_mgr;
    std::vector<std::unique_ptr<ServiceBaseInterface>> services;

    PerCoreState(uint32_t id, uint32_t heartbeat_timeout_ticks,
                 size_t timer_wheel_slots, size_t recv_buf_capacity);
};

class Server {
public:
    using Config = ServerConfig;

    explicit Server(ServerConfig config);
    ~Server();

    Server(const Server&) = delete;
    Server& operator=(const Server&) = delete;

    /// 서비스 등록 (run() 전에 호출). 체이닝 지원.
    template <typename T, typename... Args>
    Server& add_service(Args&&... args) {
        service_factories_.push_back(
            [args...](uint32_t /*core_id*/, MessageDispatcher& dispatcher)
                -> std::unique_ptr<ServiceBaseInterface> {
                auto svc = std::make_unique<T>(args...);
                svc->bind_dispatcher(dispatcher);
                return svc;
            }
        );
        return *this;
    }

    /// 블로킹 실행. 내부에서 모든 io_context, 스레드 소유.
    void run();

    /// thread-safe 종료. 시그널 핸들러 또는 다른 스레드에서 호출.
    void stop();

    /// 바인딩된 실제 포트.
    [[nodiscard]] uint16_t port() const noexcept;

    [[nodiscard]] uint32_t core_count() const noexcept;
    [[nodiscard]] bool running() const noexcept;

private:
    using ServiceFactory = std::function<
        std::unique_ptr<ServiceBaseInterface>(uint32_t, MessageDispatcher&)>;

    boost::asio::awaitable<void> read_loop(SessionPtr session, uint32_t core_id);
    boost::asio::awaitable<void> process_frames(SessionPtr session, uint32_t core_id);

    ServerConfig config_;
    boost::asio::io_context accept_io_;
    std::unique_ptr<CoreEngine> core_engine_;
    std::vector<PerCoreState> per_core_;
    MessageDispatcher dispatcher_;
    std::unique_ptr<TcpAcceptor> acceptor_;
    std::vector<ServiceFactory> service_factories_;

    std::atomic<uint32_t> next_core_{0};
    std::atomic<bool> running_{false};
    std::atomic<uint64_t> next_request_id_{0};
};

} // namespace apex::core
```

### Step 3: server.cpp 최소 구현

run()/stop() 라이프사이클만 구현. accept/read_loop는 아직 빈 상태.

```cpp
Server::Server(ServerConfig config)
    : config_(std::move(config))
{
    // CoreEngine 생성
    CoreEngineConfig engine_config{
        .num_cores = config_.num_cores,
        .mpsc_queue_capacity = config_.mpsc_queue_capacity,
        .drain_interval = config_.drain_interval,
    };
    core_engine_ = std::make_unique<CoreEngine>(engine_config);

    // PerCoreState 초기화
    per_core_.reserve(config_.num_cores);
    for (uint32_t i = 0; i < config_.num_cores; ++i) {
        per_core_.emplace_back(i, config_.heartbeat_timeout_ticks,
                               config_.timer_wheel_slots, config_.recv_buf_capacity);
    }
}

void Server::run() {
    running_.store(true);

    // 서비스 인스턴스 생성 (코어별)
    for (uint32_t core_id = 0; core_id < config_.num_cores; ++core_id) {
        for (auto& factory : service_factories_) {
            auto svc = factory(core_id, dispatcher_);
            svc->on_start();
            per_core_[core_id].services.push_back(std::move(svc));
        }
    }

    // drain 콜백 (SessionManager tick)
    core_engine_->set_drain_callback([this](uint32_t core_id) {
        if (core_id < per_core_.size()) {
            per_core_[core_id].session_mgr.tick();
        }
    });

    // CoreEngine 시작
    core_engine_->start();

    // 시그널 핸들링
    if (config_.handle_signals) {
        boost::asio::signal_set signals(accept_io_, SIGINT, SIGTERM);
        signals.async_wait([this](auto, auto) { stop(); });
        accept_io_.run();
    } else {
        accept_io_.run();
    }

    // CoreEngine 정리
    core_engine_->stop();
    core_engine_->drain_remaining();
    core_engine_->join();

    running_.store(false);
}

void Server::stop() {
    if (!running_.load()) return;

    // accept 중단
    // (TcpAcceptor가 있으면 여기서 stop)

    // accept_io_ 중단 → run() 블로킹 해제
    accept_io_.stop();
}
```

주의: 이 시점에서 TcpAcceptor, read_loop, process_frames는 미구현. Task 6, 7에서 구현.
`PerCoreState` 생성자는 `SessionManager` 생성자에 맞춰 구현할 것.

### Step 2: stop() 전체 구현

설계 문서의 5단계 시퀀스:

```cpp
void Server::stop() {
    if (running_.exchange(false) == false) return;

    // ① accept 중단
    if (acceptor_) acceptor_->stop();

    // ② 각 코어에 세션 정리 post
    std::atomic<uint32_t> remaining{config_.num_cores};
    for (uint32_t i = 0; i < config_.num_cores; ++i) {
        boost::asio::post(core_engine_->io_context(i), [this, i, &remaining] {
            per_core_[i].session_mgr.for_each([](SessionPtr session) {
                session->close();
            });
            remaining.fetch_sub(1, std::memory_order_release);
        });
    }

    // ③ 완료 대기
    while (remaining.load(std::memory_order_acquire) > 0) {
        std::this_thread::yield();
    }

    // ④ CoreEngine 중단 + 잔여 메시지 정리
    core_engine_->stop();
    core_engine_->drain_remaining();

    // ⑤ accept_io_ 중단 → run() 블로킹 해제
    accept_io_.stop();
}
```

## 리스크 및 주의사항

1. **기존 E2E 테스트 포팅 (Task 7)**: 가장 큰 작업. 모든 기존 테스트를 새 API로 변경해야 하며, ChatBroadcast 테스트는 서비스 모델 변경으로 인해 수정 필요.
2. **TcpAcceptor 인터페이스 변경 (Task 6)**: 기존 인터페이스를 확인한 뒤 최소 변경할 것. 기존 TcpAcceptor 테스트도 함께 갱신.
3. **cross_core_call 타이밍 (Task 8)**: steady_timer cancel 패턴이 Boost.Asio awaitable과 올바르게 통합되는지 주의 깊게 검증. drain_interval이 테스트 성공에 영향을 미칠 수 있으므로 50us로 설정.
4. **add_service 템플릿 캡처**: 가변 인자를 람다로 캡처하는 패턴이 MSVC에서 올바르게 동작하는지 확인. 필요 시 `std::tuple` + `std::apply`로 래핑.
